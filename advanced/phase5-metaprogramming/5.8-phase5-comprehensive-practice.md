## 2. Demo 1: 声明宏全家桶 — vec! → count! → json! DSL

### 2.1 设计哲学

> "声明宏是 Rust 的'代码模板引擎'——你描述模式，编译器填充细节。从最简单的 `vec![1,2,3]` 到能处理嵌套 JSON 的 DSL，核心原理只有三样：捕获类型、重复模式、递归展开。掌握它们，你就能写出让 Rust 代码像脚本语言一样简洁的宏。"

### 2.2 vec! 的完整手写版

```rust
/// ━━━ 5.1 macro_rules!: 捕获类型 + 重复模式 ━━━

macro_rules! my_vec {
    // ━━━ 捕获类型：expr = 表达式 ━━━
    // $($x:expr),*  →  0 个或多个以逗号分隔的表达式
    // $(,)?          →  可选的尾随逗号
    ($($x:expr),* $(,)?) => {{
        // ━━━ 5.6 const fn: Vec::with_capacity 不是 const，但 let 语句可以 ━━━
        // 预计算长度——编译期优化掉临时 Vec
        let len = 0 $(+ { let _ = &$x; 1 })*;

        let mut v = Vec::with_capacity(len);
        $(
            v.push($x);
        )*
        v
    }};
}

// ━━━ 使用 ━━━
fn demo_vec() {
    let v1 = my_vec![1, 2, 3];           // Vec<i32>
    let v2 = my_vec!["a", "b", "c",];    // 尾随逗号 OK

    // 展开后的代码（概念上）：
    // let len = 0 + 1 + 1 + 1;    // = 3
    // let mut v = Vec::with_capacity(3);
    // v.push(1); v.push(2); v.push(3);
}
```

### 2.3 count! — 递归宏 + TT Muncher

```rust
/// ━━━ 5.1 TT Muncher: 逐个消费 token ━━━
///
/// 思路：`count!(a b c d)` → `1 + count!(b c d)` → `1 + 1 + count!(c d)` → ... → `1 + 1 + 1 + 1 + 0`

macro_rules! count {
    // ━━━ 终止条件：空输入 → 0 ━━━
    () => { 0usize };

    // ━━━ 递归：消费第一个 token ($head)，计数 +1 ━━━
    // tt = token tree——最灵活的捕获类型，匹配任意单个 token
    ($head:tt $($tail:tt)*) => {
        1usize + count!($($tail)*)
    };
}

// ━━━ 展开过程（概念）━━━
// count!(a b c d)
// → 1 + count!(b c d)
// → 1 + 1 + count!(c d)
// → 1 + 1 + 1 + count!(d)
// → 1 + 1 + 1 + 1 + count!()
// → 1 + 1 + 1 + 1 + 0
// → 4 (编译期常量，零运行时开销)

fn demo_count() {
    let n = count!(x y z);  // n = 3
    assert_eq!(n, 3);
}
```

### 2.4 json! 宏 — 内部规则模式 (@标记)

```rust
/// ━━━ 5.1 内部规则模式: @ 标记分离公有接口和内部逻辑 ━━━
///
/// 这种模式在标准库中广泛使用（如 format_args!）

macro_rules! my_json {
    // ━━━ 公有接口：直接匹配 JSON 语法 ━━━
    ({}) => { String::from("{}") };

    ({ $($key:tt : $value:tt),* $(,)? }) => {{
        let mut s = String::from("{");
        $(
            s.push_str(&my_json!(@key $key));
            s.push(':');
            s.push_str(&my_json!(@value $value));
            s.push(',');
        )*
        if s.ends_with(',') { s.pop(); }
        s.push('}');
        s
    }};

    ([ $($elem:tt),* $(,)? ]) => {{
        let mut s = String::from("[");
        $(
            s.push_str(&my_json!(@value $elem));
            s.push(',');
        )*
        if s.ends_with(',') { s.pop(); }
        s.push(']');
        s
    }};

    // ━━━ 内部规则：处理不同类型的值 ━━━
    // @key: JSON key 总是字符串
    (@key $s:ident) => { stringify!($s) };
    // @value: 分派不同类型
    (@value null) => { String::from("null") };
    (@value true) => { String::from("true") };
    (@value false) => { String::from("false") };
    (@value $n:literal) => { $n.to_string() };
    (@value $s:ident) => { stringify!($s) };
}

// ━━━ 使用 ━━━
fn demo_json() {
    let json = my_json!({
        name: "Alice",
        age: 30,
        active: true,
        scores: [95, 87, 92],
        meta: {}
    });
    // → {"name":"Alice","age":30,"active":true,"scores":[95,87,92],"meta":{}}
}

// ━━━ 5.7 调试: 用 cargo expand 看展开后的代码 ━━━
// $ cargo expand --lib my_json_tests::demo_json
// 可以看到 my_json! 展开后的完整 String 构造代码
```

### 3.3 #[derive(MyDebug)]

```rust
// my_derive/src/lib.rs
use proc_macro::TokenStream;
use quote::quote;
use syn::{parse_macro_input, DeriveInput, Data};

/// ━━━ 5.2 Derive 宏: #[proc_macro_derive(MyDebug)] ━━━
#[proc_macro_derive(MyDebug)]
pub fn derive_my_debug(input: TokenStream) -> TokenStream {
    // Step 1: 解析 TokenStream → AST（syn）
    let input = parse_macro_input!(input as DeriveInput);
    let name = &input.ident;

    // Step 2: 提取字段
    let fields = match &input.data {
        Data::Struct(data) => &data.fields,
        Data::Enum(_) => {
            // ━━━ 5.4 Span 诊断: 对 enum 报友好错误 ━━━
            return syn::Error::new(
                name.span(),
                "MyDebug 暂不支持 enum，仅支持 struct",
            )
            .to_compile_error()
            .into();
        }
        Data::Union(_) => {
            return syn::Error::new(name.span(), "MyDebug 不支持 union")
                .to_compile_error()
                .into();
        }
    };

    // Step 3: 为每个字段生成 `field_name: {:?}` 的格式化字符串
    let field_names: Vec<_> = fields.iter().map(|f| &f.ident).collect();
    let format_str = field_names.iter()
        .map(|n| format!("{}: {{:?}}", n.as_ref().unwrap()))
        .collect::<Vec<_>>()
        .join(", ");
    let format_str = format!("{} {{ {} }}", name, format_str);

    // Step 4: 生成代码（quote）
    // ━━━ 5.6 const fn: format_str 编译期已知 → 生成 const &str ━━━
    let field_refs: Vec<_> = field_names.iter()
        .map(|n| quote! { &self.#n })
        .collect();

    let expanded = quote! {
        impl std::fmt::Debug for #name {
            fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
                write!(f, #format_str, #(#field_refs),*)
            }
        }
    };

    TokenStream::from(expanded)
}

// ━━━ 5.4 Span: call_site() vs def_site() ━━━
// 大多数情况用 Span::call_site() —— 指向宏调用位置
// 这确保宏展开后的编译错误指向用户的代码，而非宏定义
```

### 3.4 #[derive(Builder)] — 完整实现框架

```rust
// ━━━ 5.2 Derive 宏进阶: Builder 模式 ━━━

#[proc_macro_derive(Builder, attributes(builder))]
pub fn derive_builder(input: TokenStream) -> TokenStream {
    let input = parse_macro_input!(input as DeriveInput);
    let name = &input.ident;
    let builder_name = syn::Ident::new(&format!("{}Builder", name), name.span());

    let fields: Vec<_> = match &input.data {
        Data::Struct(data) => data.fields.iter().collect(),
        _ => return syn::Error::new(name.span(), "Builder 仅支持 struct")
            .to_compile_error().into(),
    };

    // ━━━ 5.4 Span: quote_spanned! 精确关联字段 ━━━
    // 每个生成的 setter 方法的 Span 指向原始字段——编译错误精确定位
    let setters: Vec<_> = fields.iter().map(|f| {
        let field_name = f.ident.as_ref().unwrap();
        let field_type = &f.ty;
        let setter_name = field_name;

        // ━━━ 检测 Option<T>: 自动处理可选字段 ━━━
        // 略——完整实现在 5.2 课程中

        quote_spanned! { f.span() =>
            pub fn #setter_name(&mut self, value: #field_type) -> &mut Self {
                self.#field_name = Some(value);
                self
            }
        }
    }).collect();

    let field_names: Vec<_> = fields.iter().map(|f| &f.ident).collect();
    let field_names2 = &field_names;

    let expanded = quote! {
        pub struct #builder_name {
            #(#field_names: Option<()>,)*  // 简化：实际是 Option<FieldType>
        }

        impl #builder_name {
            pub fn new() -> Self {
                #builder_name { #(#field_names2: None,)* }
            }
            #(#setters)*
            pub fn build(&mut self) -> Result<#name, String> {
                Ok(#name {
                    #(#field_names2: self.#field_names2.take().unwrap(),)*
                })
            }
        }
    };

    TokenStream::from(expanded)
}
```

### 4.3 const fn: 编译期 CRC32 查找表

```rust
/// ━━━ 5.6 const fn: 编译期计算 256 个 CRC32 表项 ━━━
///
/// 传统做法：在运行时生成 CRC32 表（256 * 4 = 1024 字节）
/// const fn 做法：在编译期生成——表项在二进制中就绪，零初始化开销

const fn compute_crc32_table() -> [u32; 256] {
    let mut table = [0u32; 256];
    let mut i = 0;
    while i < 256 {
        let mut crc = i as u32;
        let mut j = 0;
        while j < 8 {
            if crc & 1 != 0 {
                crc = (crc >> 1) ^ 0xEDB88320;
            } else {
                crc >>= 1;
            }
            j += 1;
        }
        table[i] = crc;
        i += 1;
    }
    table
}

/// ━━━ 5.1 macro_rules! 辅助: 用宏简化 CRC32 调用 ━━━
const CRC32_TABLE: [u32; 256] = compute_crc32_table();

pub const fn crc32(data: &[u8]) -> u32 {
    let mut crc = 0xFFFFFFFFu32;
    let mut i = 0;
    while i < data.len() {
        let idx = ((crc ^ data[i] as u32) & 0xFF) as usize;
        crc = (crc >> 8) ^ CRC32_TABLE[idx];
        i += 1;
    }
    crc ^ 0xFFFFFFFF
}

// ━━━ 编译期 CRC32: 零运行时开销（结果嵌入二进制）━━━
const FILE_MAGIC: u32 = crc32(b"MYFILE");

// ━━━ 5.4 编译期诊断: 静态断言（5.6）━━━
// 在编译期验证 CRC32 值是否匹配预期
const _: () = assert!(FILE_MAGIC != 0, "CRC32 不能为 0");

fn demo_const_crc() {
    // 运行时也可以调用——同样的 const fn，两种上下文中使用
    let runtime_crc = crc32(b"hello world");
    println!("CRC32: {:#x}", runtime_crc);
}
```

### 4.4 const generics: 编译期维度检查的矩阵

```rust
/// ━━━ 5.6 const generics: 编译期保证矩阵维度匹配 ━━━
///
/// 不匹配的矩阵乘法在编译期就报错——不是运行时 panic

#[derive(Debug, Clone)]
pub struct Matrix<T, const ROWS: usize, const COLS: usize> {
    data: [[T; COLS]; ROWS],
}

impl<T: Copy + Default, const R: usize, const C: usize> Matrix<T, R, C> {
    pub const fn new() -> Self {
        Matrix { data: [[T::default(); C]; R] }
    }

    pub fn from_array(data: [[T; C]; R]) -> Self {
        Matrix { data }
    }
}

/// ━━━ 矩阵乘法: const generics 编译期检查维度 ━━━
/// Matrix<R, K> * Matrix<K, C> = Matrix<R, C>
impl<T, const R: usize, const K: usize, const C: usize>
    std::ops::Mul<Matrix<T, K, C>> for Matrix<T, R, K>
where
    T: Copy + std::ops::Add<Output = T> + std::ops::Mul<Output = T> + Default,
{
    type Output = Matrix<T, R, C>;

    fn mul(self, rhs: Matrix<T, K, C>) -> Self::Output {
        let mut result = Matrix::new();
        for i in 0..R {
            for j in 0..C {
                let mut sum = T::default();
                #[allow(clippy::needless_range_loop)]
                for k in 0..K {
                    sum = sum + self.data[i][k] * rhs.data[k][j];
                }
                result.data[i][j] = sum;
            }
        }
        result
    }
}

fn demo_matrix() {
```

### 5.2 #[log_call] — Attribute 宏注入日志

```rust
// ━━━ 5.3 Attribute 宏: #[proc_macro_attribute] ━━━
///
/// attr: TokenStream = 宏的参数（括号里的内容）
/// item: TokenStream = 被标注的函数/结构体/等
use proc_macro::TokenStream;
use quote::quote;
use syn::{parse_macro_input, ItemFn};

#[proc_macro_attribute]
pub fn log_call(attr: TokenStream, item: TokenStream) -> TokenStream {
    // ━━━ 解析 attr 参数 ━━━
    let level = if attr.is_empty() {
        "INFO".to_string()
    } else {
        let level_str = attr.to_string();
        level_str.trim_matches('"').to_string()
    };

    // ━━━ 解析被标注的函数 ━━━
    let input_fn = parse_macro_input!(item as ItemFn);
    let fn_name = &input_fn.sig.ident;
    let fn_block = &input_fn.block;
    let fn_vis = &input_fn.vis;
    let fn_sig = &input_fn.sig;

    // ━━━ 5.4 Span: quote_spanned! 保持原始 Span ━━━
    // 生成的代码的 Span 指向原始函数——编译错误定位到用户代码
    let expanded = quote_spanned! { input_fn.span() =>
        #fn_vis #fn_sig {
            // ━━━ 注入的日志代码 ━━━
            println!(
                "[{}] → {}() 被调用 at {}:{}",
                #level,
                stringify!(#fn_name),
                file!(),
                line!(),
            );

            // ━━━ 原始函数体 ━━━
            #fn_block
        }
    };

    TokenStream::from(expanded)
}

// ━━━ 使用 ━━━
// #[log_call("DEBUG")]
// fn calculate(x: i32, y: i32) -> i32 {
//     x + y
// }
//
// 展开后：
// fn calculate(x: i32, y: i32) -> i32 {
//     println!("[DEBUG] → calculate() 被调用 at src/main.rs:42");
//     { x + y }
```

### 5.3 sql! — Function-like 宏（编译期 SQL 校验）

```rust
/// ━━━ 5.3 Function-like 宏: #[proc_macro] ━━━
///
/// sql!("SELECT * FROM users WHERE id = ?")  → 编译期语法校验
#[proc_macro]
pub fn sql(input: TokenStream) -> TokenStream {
    let sql_str = input.to_string();
    let sql_str = sql_str.trim_matches('"');

    // ━━━ 编译期 SQL 校验 ━━━
    let validation = validate_sql(sql_str);
    if let Err(err) = validation {
        // ━━━ 5.4 Span 诊断: 多部分错误信息 ━━━
        let mut diag = syn::Error::new(
            proc_macro2::Span::call_site(),
            format!("SQL 语法错误: {}", err),
        );
        diag.combine(syn::Error::new(
            proc_macro2::Span::call_site(),
            "提示: 检查关键字拼写（SELECT/FROM/WHERE）",
        ));
        return diag.to_compile_error().into();
    }

    // ━━━ 生成代码: 返回预编译的 SQL 字符串 ━━━
    let expanded = quote! {
        // ━━━ 5.6 const fn: SQL 字符串编译期已知 ━━━
        SqlQuery {
            sql: #sql_str,
            params: Vec::new(),
        }
    };

    TokenStream::from(expanded)
}

fn validate_sql(sql: &str) -> Result<(), String> {
    let upper = sql.to_uppercase();
    if !upper.contains("SELECT") && !upper.contains("INSERT")
        && !upper.contains("UPDATE") && !upper.contains("DELETE")
    {
        return Err("缺少 DML 关键字（SELECT/INSERT/UPDATE/DELETE）".into());
    }
    Ok(())
}

struct SqlQuery {
    sql: &'static str,
    params: Vec<String>,
}
```

### 6.1 题目

实现一个迷你序列化框架，组合使用 Phase 5 所有技术：

1. **`#[derive(MySerialize)]`** (5.2) — 为 struct 自动生成 `serialize()` → `String`
2. **`#[serde(rename = "...")]`** attribute (5.3) — 字段属性：自定义序列化名
3. **`my_json!()` 宏** (5.1) — 用声明宏构造 JSON 字面量
4. **Span 诊断** (5.4) — 对 enum/union 报友好错误
5. **trybuild 测试** (5.4, 5.7) — pass + fail 用例
6. **编译期验证** (5.6) — `const _: () = assert!` 验证生成的代码

### 6.2 参考答案

```rust
// ━━━ Step 1: derive MySerialize ━━━
use proc_macro::TokenStream;
use syn::{parse_macro_input, DeriveInput, Data, Fields, Meta};
use quote::{quote, quote_spanned};

#[proc_macro_derive(MySerialize, attributes(serde))]
pub fn derive_my_serialize(input: TokenStream) -> TokenStream {
    let input = parse_macro_input!(input as DeriveInput);
    let name = &input.ident;

    let fields = match &input.data {
        Data::Struct(data) => match &data.fields {
            Fields::Named(fields) => &fields.named,
            _ => return syn::Error::new(name.span(), "仅支持具名字段 struct")
                .to_compile_error().into(),
        },
        _ => return syn::Error::new(name.span(), "仅支持 struct")
            .to_compile_error().into(),
    };

    // ━━━ 提取 #[serde(rename = "...")] ━━━
    let serializations: Vec<_> = fields.iter().map(|f| {
        let field_name = f.ident.as_ref().unwrap();
        let field_name_str = field_name.to_string();

        // 查找 rename 属性
        let rename = f.attrs.iter()
            .filter(|a| a.path().is_ident("serde"))
            .find_map(|a| {
                a.parse_args::<syn::MetaNameValue>().ok()
                    .filter(|nv| nv.path.is_ident("rename"))
                    .and_then(|nv| {
                        if let syn::Expr::Lit(lit) = &nv.value {
                            if let syn::Lit::Str(s) = &lit.lit {
                                return Some(s.value());
                            }
                        }
                        None
                    })
            })
            .unwrap_or(field_name_str);

        quote_spanned! { f.span() =>
            s.push_str(#rename);
            s.push(':');
            // 递归序列化值
            s.push_str(&self.#field_name.to_string());
            s.push(',');
        }
    }).collect();

    let expanded = quote! {
        impl MySerialize for #name {
            fn serialize(&self) -> String {
                let mut s = String::from("{");
                #(#serializations)*
                if s.ends_with(',') { s.pop(); }
                s.push('}');
                s
            }
        }
    };

    TokenStream::from(expanded)
}

pub trait MySerialize {
    fn serialize(&self) -> String;
}
```

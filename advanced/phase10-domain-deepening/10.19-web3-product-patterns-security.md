## 一、链上程序架构模式

### 1.1 单程序 + 多数据账户

这是 Solana 最基础的架构模式：一个程序控制多个独立的数据账户。

```rust
// 单程序 + 多数据账户模式 (Anchor 实现)

use anchor_lang::prelude::*;

declare_id!("Fg6PaFpoGXkYsidMpWTK6W2BeZ7FEfcYkg476zPFsLnS");

// ---- 数据账户: 每个用户一个 ----
#[account]
pub struct UserProfile {
    pub owner: Pubkey,           // 32 bytes
    pub username: String,        // 4 + len
    pub bio: String,             // 4 + len
    pub avatar_url: String,      // 4 + len
    pub reputation: u64,         // 8 bytes
    pub created_at: i64,         // 8 bytes
    pub bump: u8,                // 1 byte (PDA bump)
}

// ---- 全局配置账户: 整个程序一个 ----
#[account]
pub struct GlobalConfig {
    pub authority: Pubkey,           // 谁可以修改配置
    pub fee_basis_points: u16,       // 手续费 (基点)
    pub paused: bool,                // 紧急暂停开关
    pub total_users: u64,            // 总用户数 (统计用途)
    pub bump: u8,
}

// ---- 指令处理器 ----

#[program]
pub mod user_profile_program {
    use super::*;

    /// 初始化全局配置 (只调用一次)
    pub fn initialize(ctx: Context<Initialize>, fee_bps: u16) -> Result<()> {
        let config = &mut ctx.accounts.config;
        config.authority = ctx.accounts.authority.key();
        config.fee_basis_points = fee_bps;
        config.paused = false;
        config.total_users = 0;
        config.bump = ctx.bumps.config;
        Ok(())
    }

    /// 创建用户资料
    pub fn create_profile(
        ctx: Context<CreateProfile>,
        username: String,
        bio: String,
        avatar_url: String,
    ) -> Result<()> {
        require!(!ctx.accounts.config.paused, ErrorCode::ProgramPaused);

        let profile = &mut ctx.accounts.profile;
        profile.owner = ctx.accounts.user.key();
        profile.username = username;
        profile.bio = bio;
        profile.avatar_url = avatar_url;
        profile.reputation = 0;
        profile.created_at = Clock::get()?.unix_timestamp;
        profile.bump = ctx.bumps.profile;

        // 更新全局计数
        ctx.accounts.config.total_users += 1;

        Ok(())
    }

    /// 更新用户资料 (只有 owner 可以)
    pub fn update_profile(
        ctx: Context<UpdateProfile>,
        new_bio: String,
    ) -> Result<()> {
        let profile = &mut ctx.accounts.profile;
        profile.bio = new_bio;
        Ok(())
    }
}

// ---- 账户验证结构 ----

#[derive(Accounts)]
pub struct Initialize<'info> {
    #[account(mut)]
    pub authority: Signer<'info>,
    #[account(
        init,
        payer = authority,
        space = 8 + 32 + 2 + 1 + 8 + 1,  // discriminator + fields
        seeds = [b"global_config"],
        bump,
    )]
    pub config: Account<'info, GlobalConfig>,
    pub system_program: Program<'info, System>,
}

#[derive(Accounts)]
pub struct CreateProfile<'info> {
    #[account(mut)]
    pub user: Signer<'info>,
    #[account(
        init,
        payer = user,
        space = 8 + 32 + (4 + 50) + (4 + 200) + (4 + 100) + 8 + 8 + 1,
        seeds = [b"profile", user.key().as_ref()],
        bump,
    )]
    pub profile: Account<'info, UserProfile>,
    #[account(
        seeds = [b"global_config"],
        bump = config.bump,
    )]
    pub config: Account<'info, GlobalConfig>,
    pub system_program: Program<'info, System>,
}

#[derive(Accounts)]
pub struct UpdateProfile<'info> {
    #[account(mut)]
    pub user: Signer<'info>,
    #[account(
        mut,
        seeds = [b"profile", user.key().as_ref()],
        bump = profile.bump,
        constraint = profile.owner == user.key() @ ErrorCode::Unauthorized,
    )]
    pub profile: Account<'info, UserProfile>,
}

#[error_code]
pub enum ErrorCode {
    #[msg("Program is paused")]
    ProgramPaused,
    #[msg("Unauthorized access")]
    Unauthorized,
}
```

**模式总结**：

```
┌──────────────────────────────────────┐
│           Program (不可变)            │
│  指令处理器: initialize, create, update │
├──────────────────────────────────────┤
│         GlobalConfig (PDA)            │
│  authority, fee, paused, total_users  │
├───────────┬───────────┬──────────────┤
│ Profile_A │ Profile_B │ Profile_C    │
│ (PDA)    │ (PDA)    │ (PDA)         │
│ owner: A │ owner: B │ owner: C      │
└───────────┴───────────┴──────────────┘
```

### 1.2 程序工厂模式

```rust
// 程序工厂模式: 一个程序创建另一个程序的实例

// 场景: DEX 工厂合约 -> 为每个交易对创建独立的流动性池

use anchor_lang::prelude::*;
use anchor_spl::token::{Mint, TokenAccount};

declare_id!("DexFactory11111111111111111111111111111111111111");

#[program]
pub mod dex_factory {
    use super::*;

    /// 创建新的交易对池
    pub fn create_pool(
        ctx: Context<CreatePool>,
        initial_price: u64,
        fee_bps: u16,
    ) -> Result<()> {
        let pool = &mut ctx.accounts.pool;
        pool.factory = ctx.accounts.factory.key();
        pool.token_a = ctx.accounts.token_a.key();
        pool.token_b = ctx.accounts.token_b.key();
        pool.reserve_a = 0;
        pool.reserve_b = 0;
        pool.fee_basis_points = fee_bps;
        pool.lp_supply = 0;
        pool.bump = ctx.bumps.pool;

        // 发出事件
        emit!(PoolCreated {
            pool: pool.key(),
            token_a: pool.token_a,
            token_b: pool.token_b,
            initial_price,
            fee_bps,
        });

        Ok(())
    }
}

#[derive(Accounts)]
pub struct CreatePool<'info> {
    #[account(mut)]
    pub payer: Signer<'info>,

    /// 工厂 PDA (记录所有创建的池)
    #[account(
        seeds = [b"dex_factory"],
        bump,
    )]
    pub factory: Account<'info, FactoryState>,

    /// 新池的 PDA (种子由 token 对决定)
    #[account(
        init,
        payer = payer,
        space = 8 + PoolState::LEN,
        seeds = [
            b"pool",
            token_a.key().as_ref(),
            token_b.key().as_ref(),
        ],
        bump,
    )]
    pub pool: Account<'info, PoolState>,

    pub token_a: Account<'info, Mint>,
    pub token_b: Account<'info, Mint>,

    pub system_program: Program<'info, System>,
}

#[account]
pub struct FactoryState {
    pub authority: Pubkey,
    pub pool_count: u64,
    pub total_volume: u64,
    pub bump: u8,
}

#[account]
pub struct PoolState {
    pub factory: Pubkey,
    pub token_a: Pubkey,
    pub token_b: Pubkey,
    pub reserve_a: u64,
    pub reserve_b: u64,
    pub fee_basis_points: u16,
    pub lp_supply: u64,
    pub bump: u8,
}

impl PoolState {
    const LEN: usize = 32 + 32 + 32 + 8 + 8 + 2 + 8 + 1;
}

#[event]
pub struct PoolCreated {
    pub pool: Pubkey,
    pub token_a: Pubkey,
    pub token_b: Pubkey,
    pub initial_price: u64,
    pub fee_bps: u16,
}

fn factory_pattern_summary() {
    println!("=== 程序工厂模式 ===\n");
    println!("适用场景:");
    println!("  • DEX: 为每个交易对创建独立的流动性池");
    println!("  • NFT 市场: 为每个集合创建独立的 marketplace");
    println!("  • 借贷协议: 为每种资产创建独立的借贷池");
    println!("  • DAO: 为每个组织创建独立的治理实例\n");
    println!("PDA 种子设计:");
    println!("  • 包含区分实例的标识符 (token pair, collection name)");
    println!("  • 确保每个实例的 PDA 唯一且可预测");
    println!("  • 其他程序可以通过相同的种子计算找到任何池");
}
```

```rust
// 核心思想:

fn explain_proxy_pattern() {

// Anchor 程序中的升级设计
#[program]
pub mod upgradeable_contract {
    use super::*;

    // V1 版本的初始化
    pub fn initialize_v1(ctx: Context<Initialize>, admin: Pubkey) -> Result<()> {
        let state = &mut ctx.accounts.state;
        state.admin = admin;
        state.version = 1;
        state.data_v1 = 0;
        state.reserved = [0u8; 256];  // 预留空间给未来版本
        state.bump = ctx.bumps.state;
        Ok(())
    }

    // V2 版本的迁移函数
    pub fn migrate_to_v2(ctx: Context<MigrateV2>) -> Result<()> {
        let state = &mut ctx.accounts.state;
        require!(state.version == 1, ErrorCode::InvalidVersion);

        // 从预留空间分配 v2 字段
        state.version = 2;
        // data_v2 从 reserved[0..8] 读写
        Ok(())
    }
}

#[account]
pub struct ContractState {
    pub admin: Pubkey,          // 32
    pub version: u8,            // 1
    pub data_v1: u64,           // 8
    pub reserved: [u8; 256],    // 256 (预留给未来扩展)
    pub bump: u8,               // 1
}

#[derive(Accounts)]
pub struct MigrateV2<'info> {
    #[account(mut, constraint = state.admin == admin.key() @ ErrorCode::Unauthorized)]
    pub admin: Signer<'info>,
    #[account(mut, seeds = [b"state"], bump = state.bump)]
    pub state: Account<'info, ContractState>,
}

#[error_code]
pub enum ErrorCode {
    #[msg("Invalid version for migration")]
    InvalidVersion,
    #[msg("Unauthorized")]
    Unauthorized,
}
```

## 二、安全审计方法论

### 2.1 Solana 常见漏洞分类

```rust
// ============================================
// 漏洞1: 缺少签名者检查
// ============================================

// ❌ 危险: 任何人都可以调用, 没有检查调用者身份
pub fn dangerous_update_admin(ctx: Context<DangerousUpdate>, new_admin: Pubkey) -> Result<()> {
    ctx.accounts.config.admin = new_admin;  // 任何人都可以修改 admin!
    Ok(())
}

#[derive(Accounts)]
pub struct DangerousUpdate<'info> {
    #[account(mut)]
    pub config: Account<'info, GlobalConfig>,
    // ❌ 缺少 Signer 检查! 任何人都可以传入任意账户
}

// ✅ 安全: 验证签名者
#[derive(Accounts)]
pub struct SafeUpdate<'info> {
    pub authority: Signer<'info>,  // ✅ 必须有签名者
    #[account(
        mut,
        constraint = config.admin == authority.key() @ ErrorCode::Unauthorized
        // ✅ 验证签名者是当前 admin
    )]
    pub config: Account<'info, GlobalConfig>,
}


// ============================================
// 漏洞2: 缺少所有者检查
// ============================================

// ❌ 危险: 没有检查账户所属的程序
pub fn dangerous_token_op(ctx: Context<DangerousToken>) -> Result<()> {
    // 直接操作 token_account, 但这是否真的是 Token Program 的账户?
    let _amount = ctx.accounts.token_account.amount;
    Ok(())
}

#[derive(Accounts)]
pub struct DangerousToken<'info> {
    /// CHECK: ❌ 没有验证 owner!
    pub token_account: AccountInfo<'info>,
}

// ✅ 安全: 验证账户所有者和类型
#[derive(Accounts)]
pub struct SafeToken<'info> {
    #[account(
        // ✅ Anchor 自动验证: 属于 Token Program 的 TokenAccount 类型
        constraint = token_account.owner == user.key(),
    )]
    pub token_account: Account<'info, TokenAccount>,
    pub user: Signer<'info>,
}


// ============================================
// 漏洞3: PDA 种子未验证
// ============================================

// ❌ 危险: PDA 种子包含用户提供的未验证数据
pub fn dangerous_pda(ctx: Context<DangerousPda>, fake_key: Pubkey) -> Result<()> {
    msg!("PDA: {}", ctx.accounts.vault.key());
    // 如果 fake_key 是恶意的, PDA 就不是预期的地址!
    Ok(())
}

#[derive(Accounts)]
pub struct DangerousPda<'info> {
    #[account(
        mut,
        seeds = [b"vault", fake_key.as_ref()],  // ❌ 用未验证的输入作为种子!
        bump,
    )]
    pub vault: Account<'info, TokenAccount>,
    /// CHECK: 用户提供的 key, 可能指向攻击者的账户
    pub fake_key: AccountInfo<'info>,
}

// ✅ 安全: 验证种子来源
#[derive(Accounts)]
pub struct SafePda<'info> {
    pub user: Signer<'info>,
    #[account(
        mut,
        seeds = [b"vault", user.key().as_ref()],  // ✅ 从已验证的账户派生
        bump,
    )]
    pub vault: Account<'info, TokenAccount>,
}


// ============================================
// 漏洞4: 账户数据未初始化检查 (Reinitialization)
// ============================================

// ❌ 危险: 没有检查账户是否已经初始化
pub fn dangerous_init(ctx: Context<DangerousInit>) -> Result<()> {
    let state = &mut ctx.accounts.state;
    state.authority = ctx.accounts.user.key();
    state.initialized = true;
    // 如果 state 之前已初始化, 攻击者可以覆盖 authority!
    Ok(())
}

#[derive(Accounts)]
pub struct DangerousInit<'info> {
    #[account(mut)]  // ❌ 没有 init 约束!
    pub state: Account<'info, ContractState>,
    #[account(mut)]
    pub user: Signer<'info>,
}

// ✅ 安全: 区分初始化和更新
pub fn safe_init(ctx: Context<SafeInit>) -> Result<()> {
    let state = &mut ctx.accounts.state;
    state.authority = ctx.accounts.user.key();
    state.initialized = true;
    state.bump = ctx.bumps.state;
    Ok(())
}

#[derive(Accounts)]
pub struct SafeInit<'info> {
    #[account(
        init,  // ✅ Anchor 保证只初始化一次
        payer = user,
        space = 8 + ContractState::LEN,
        seeds = [b"state", user.key().as_ref()],
        bump,
    )]
    pub state: Account<'info, ContractState>,
    #[account(mut)]
    pub user: Signer<'info>,
    pub system_program: Program<'info, System>,
}

// 如果逻辑上需要"可能已初始化"的情况:
pub fn maybe_init_or_update(ctx: Context<MaybeInit>) -> Result<()> {
    require!(
        ctx.accounts.state.initialized,
        ErrorCode::NotInitialized
    );
    // 做更新操作...
    Ok(())
}

#[derive(Accounts)]
pub struct MaybeInit<'info> {
    #[account(mut)]
    pub state: Account<'info, ContractState>,
    pub authority: Signer<'info>,
}


// ============================================
// 漏洞5: 整数溢出/下溢
// ============================================

// ❌ 危险: 未检查的算术运算
pub fn dangerous_arithmetic(ctx: Context<DangerousArithmetic>, amount: u64) -> Result<()> {
    let vault = &mut ctx.accounts.vault;
    vault.balance = vault.balance - amount;  // ❌ 可能下溢 Panic!
    vault.total_withdrawn = vault.total_withdrawn + amount;  // ❌ 可能溢出
    Ok(())
}

// ✅ 安全: 显式检查
pub fn safe_arithmetic(ctx: Context<SafeArithmetic>, amount: u64) -> Result<()> {
    let vault = &mut ctx.accounts.vault;
    require!(vault.balance >= amount, ErrorCode::InsufficientBalance);

    vault.balance = vault.balance.checked_sub(amount)
        .ok_or(ErrorCode::ArithmeticUnderflow)?;

    vault.total_withdrawn = vault.total_withdrawn.checked_add(amount)
        .ok_or(ErrorCode::ArithmeticOverflow)?;

    Ok(())
}

// ✅ 安全: 加入手续费和最小输出防护
pub fn safe_swap(
    amount_in: u64,
    reserve_in: u64,
    reserve_out: u64,
    fee_bps: u16,       // 手续费 (基点)
    min_amount_out: u64, // 滑点保护
) -> Result<u64> {
    let amount_in_after_fee = amount_in
        .checked_mul(10_000 - fee_bps as u64)
        .ok_or(ErrorCode::ArithmeticOverflow)?
        .checked_div(10_000)
        .ok_or(ErrorCode::ArithmeticUnderflow)?;

    let numerator = amount_in_after_fee
        .checked_mul(reserve_out)
        .ok_or(ErrorCode::ArithmeticOverflow)?;

    let amount_out = numerator
        .checked_div(reserve_in + amount_in_after_fee)
        .ok_or(ErrorCode::ArithmeticUnderflow)?;

    require!(amount_out >= min_amount_out, ErrorCode::SlippageExceeded);

    Ok(amount_out)
}
```

```rust
    fn new() -> Self {
        let mut engine = AuditEngine { rules: Vec::new() };

        // 内置规则: 检查缺失 Signer 检查
        engine.add_rule(AuditRule {
            id: "SOL-001".into(),
            severity: Severity::Critical,
            category: "Access Control".into(),
            description: "检查所有写操作的指令是否有 Signer 验证".into(),
            check_fn: Box::new(|ctx: &AuditContext| -> Vec<AuditFinding> {
                let mut findings = Vec::new();
                for instr in &ctx.instructions {
                    // 如果指令修改了账户状态但没有签名者检查
                    if !instr.has_signer_check && !instr.name.starts_with("get_") {
                        findings.push(AuditFinding {
                            rule_id: "SOL-001".into(),
                            severity: Severity::Critical,
                            location: instr.name.clone(),
                            description: "写操作缺少 Signer 检查".into(),
                            recommendation: "添加 Signer 账户并验证调用者权限".into(),
                        });
                    }
                }
                findings
            }),
        });

        // 内置规则: 检查未初始化写入
        engine.add_rule(AuditRule {
            id: "SOL-002".into(),
            severity: Severity::High,
            category: "Account Validation".into(),
            description: "检查可变账户是否有初始化保护".into(),
            check_fn: Box::new(|ctx: &AuditContext| -> Vec<AuditFinding> {
                let mut findings = Vec::new();
                for acc in &ctx.accounts {
                    if !acc.has_init_constraint && acc.account_type != "system" {
                        findings.push(AuditFinding {
                            rule_id: "SOL-002".into(),
                            severity: Severity::High,
                            location: acc.name.clone(),
                            description: "可变账户缺少初始化约束".into(),
                            recommendation: "添加 #[account(init)] 或 manual initialized check".into(),
                        });
                    }
                }
                findings
            }),
        });

        // 更多规则...
        engine
    }

    fn add_rule(&mut self, rule: AuditRule) {
        self.rules.push(rule);
    }

    /// 执行完整审计
    fn audit(&self, ctx: &AuditContext) -> Vec<AuditFinding> {
        let mut all_findings = Vec::new();
        for rule in &self.rules {
            let findings = (rule.check_fn)(ctx);
            all_findings.extend(findings);
        }

        // 按严重性排序
        all_findings.sort_by(|a, b| {
            let severity_order = |s: &Severity| match s {
                Severity::Critical => 0,
                Severity::High => 1,
                Severity::Medium => 2,
                Severity::Low => 3,
                Severity::Info => 4,
            };
            severity_order(&a.severity).cmp(&severity_order(&b.severity))
        });

        all_findings
    }

    /// 生成审计报告
    fn generate_report(&self, findings: &[AuditFinding]) -> String {
        let mut report = String::from("=== Solana 安全审计报告 ===\n\n");

        // 统计
        let critical = findings.iter().filter(|f| f.severity == Severity::Critical).count();
        let high = findings.iter().filter(|f| f.severity == Severity::High).count();
        let medium = findings.iter().filter(|f| f.severity == Severity::Medium).count();
        let low = findings.iter().filter(|f| f.severity == Severity::Low).count();

        report.push_str(&format!(
            "发现汇总: Critical={}, High={}, Medium={}, Low={}\n\n",
            critical, high, medium, low,
        ));

        for finding in findings {
            report.push_str(&format!(
                "[{:?}] {}: {}\n  建议: {}\n\n",
                finding.severity,
                finding.location,
                finding.description,
                finding.recommendation,
            ));
        }

        report
    }
}
```

```rust
// 代币归属 (Vesting): 锁定期 + 线性释放

struct VestingSchedule {
    beneficiary: Pubkey,        // 受益人
    total_amount: u64,          // 总归属额
    start_time: i64,            // 开始时间
    cliff_duration: i64,        // 悬崖期 (秒) - 在此之前无释放
    vesting_duration: i64,      // 线性释放期 (秒)
    released_amount: u64,       // 已释放额
}

impl VestingSchedule {
    /// 计算当前可领取金额
    fn available_to_claim(&self, current_time: i64) -> u64 {
        // 悬崖期: 如果还没到悬崖结束，可领取为 0
        let cliff_end = self.start_time + self.cliff_duration;
        if current_time < cliff_end {
            return 0;
        }

        // 线性释放期结束: 全部解锁
        let vesting_end = cliff_end + self.vesting_duration;
        if current_time >= vesting_end {
            return self.total_amount - self.released_amount;
        }

        // 线性释放: amount = total * (elapsed / duration) - released
        let elapsed = (current_time - cliff_end) as u64;
        let duration = self.vesting_duration as u64;
        let vested = self.total_amount
            .saturating_mul(elapsed)
            .saturating_div(duration);

        vested.saturating_sub(self.released_amount)
    }

    /// 执行领取
    fn claim(&mut self, current_time: i64) -> Result<u64, &'static str> {
        let amount = self.available_to_claim(current_time);
        if amount == 0 {
            return Err("Nothing to claim");
        }
        self.released_amount = self.released_amount.checked_add(amount)
            .ok_or("Overflow")?;
        Ok(amount)
    }
}
```

### 3.3 AMM 与流动性

```rust
// AMM 常数乘积公式: x * y = k

struct ConstantProductPool {
    reserve_x: u64,     // 代币 X 储备量
    reserve_y: u64,     // 代币 Y 储备量
    lp_supply: u64,     // LP 代币总供应量
    fee_bps: u16,       // 手续费 (基点, 如 30 = 0.3%)
}

impl ConstantProductPool {
    /// 交换: amount_x_in -> amount_y_out
    fn swap_x_for_y(&mut self, amount_x_in: u64) -> Result<u64, &'static str> {
        // 手续费
        let amount_in_after_fee = amount_x_in
            .checked_mul((10_000 - self.fee_bps) as u64)
            .ok_or("Overflow")?
            .checked_div(10_000)
            .ok_or("Division error")?;

        // 常数乘积: (x + dx) * (y - dy) = x * y
        // => dy = y * dx / (x + dx)
        let numerator = self.reserve_y
            .checked_mul(amount_in_after_fee)
            .ok_or("Overflow")?;

        let denominator = self.reserve_x
            .checked_add(amount_in_after_fee)
            .ok_or("Overflow")?;

        let amount_y_out = numerator
            .checked_div(denominator)
            .ok_or("Division error")?;

        // 更新储备
        self.reserve_x = self.reserve_x.checked_add(amount_in_after_fee).unwrap();
        self.reserve_y = self.reserve_y.checked_sub(amount_y_out).unwrap();

        // 更新 k 值 (因手续费的缘故 k 会增加)
        // k_new = (x+dx)*(y-dy) > x*y (手续费留在池中)

        Ok(amount_y_out)
    }

    /// 添加流动性: 返回 LP 代币数量
    fn add_liquidity(
        &mut self,
        amount_x: u64,
        amount_y: u64,
    ) -> Result<u64, &'static str> {
        // 如果池子为空 (初始流动性)
        if self.lp_supply == 0 {
            let lp = ((amount_x as u128 * amount_y as u128) as f64).sqrt() as u64;
            self.reserve_x = amount_x;
            self.reserve_y = amount_y;
            self.lp_supply = lp;
            return Ok(lp);
        }

        // 按比例增加流动性
        // lp = min(amount_x * lp_supply / reserve_x, amount_y * lp_supply / reserve_y)
        let lp_from_x = amount_x
            .checked_mul(self.lp_supply)
            .ok_or("Overflow")?
            .checked_div(self.reserve_x)
            .ok_or("Division error")?;

        let lp_from_y = amount_y
            .checked_mul(self.lp_supply)
            .ok_or("Overflow")?
            .checked_div(self.reserve_y)
            .ok_or("Division error")?;

        let lp_minted = lp_from_x.min(lp_from_y);

        self.reserve_x = self.reserve_x.checked_add(amount_x).unwrap();
        self.reserve_y = self.reserve_y.checked_add(amount_y).unwrap();
        self.lp_supply = self.lp_supply.checked_add(lp_minted).unwrap();

        Ok(lp_minted)
    }

    /// 移除流动性
    fn remove_liquidity(&mut self, lp_amount: u64) -> Result<(u64, u64), &'static str> {
        let share = lp_amount as f64 / self.lp_supply as f64;
        let amount_x = (self.reserve_x as f64 * share) as u64;
        let amount_y = (self.reserve_y as f64 * share) as u64;

        self.reserve_x = self.reserve_x.checked_sub(amount_x).unwrap();
        self.reserve_y = self.reserve_y.checked_sub(amount_y).unwrap();
        self.lp_supply = self.lp_supply.checked_sub(lp_amount).unwrap();

        Ok((amount_x, amount_y))
    }
}
```

```rust
    success: bool,
    data: Option<T>,
    error: Option<String>,

impl<T: Serialize> ApiResponse<T> {
    fn ok(data: T) -> Self {
        ApiResponse { success: true, data: Some(data), error: None }
    }

    fn err(msg: impl Into<String>) -> Self {
        ApiResponse { success: false, data: None, error: Some(msg.into()) }
    }
}

async fn get_block(

    Path(slot): Path<u64>,
) -> Json<ApiResponse<BlockInfo>> {
    match state.blockchain.read().await.get_block(slot).await {
        Ok(block) => Json(ApiResponse::ok(block)),
        Err(e) => Json(ApiResponse::err(e)),
    }
}

async fn get_balance(
    State(state): State<AppState>,
    Path(address): Path<String>,
) -> Json<ApiResponse<AccountInfo>> {
    match state.blockchain.read().await.get_balance(&address).await {
        Ok(balance) => Json(ApiResponse::ok(AccountInfo { address, balance })),
        Err(e) => Json(ApiResponse::err(e)),
    }
}

async fn send_transaction(
    State(state): State<AppState>,
    Json(req): Json<SendTxRequest>,
) -> Json<ApiResponse<String>> {
    match state.blockchain.read().await.send_transaction(&req.transaction).await {
        Ok(sig) => Json(ApiResponse::ok(sig)),
        Err(e) => Json(ApiResponse::err(e)),
    }
}
```

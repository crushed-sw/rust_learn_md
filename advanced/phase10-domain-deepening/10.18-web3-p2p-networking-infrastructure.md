## 一、rust-libp2p：模块化 P2P 网络栈

### 1.1 libp2p 架构概览

libp2p 是 IPFS/Filecoin/Ethereum 2.0/Polkadot 等项目使用的模块化 P2P 网络框架。它的设计哲学是"每个组件都可替换"。

```
┌──────────────────────────────────────────┐
│              Application                 │
│  (你的业务逻辑: 聊天 / 区块同步 / 文件共享)   │
├──────────────────────────────────────────┤
│  Pub/Sub  │  Kademlia DHT  │  Request/   │
│ (GossipSub│  (节点发现)     │  Response   │
├──────────────────────────────────────────┤
│  Stream Muxing (yamux / mplex)           │
├──────────────────────────────────────────┤
│  Transport (TCP / QUIC / WebSocket / ...) │
├──────────────────────────────────────────┤
│  Security (Noise / TLS 1.3)              │
├──────────────────────────────────────────┤
│  Peer Identity (PeerId from public key)  │
└──────────────────────────────────────────┘
```

```rust
use libp2p::{
    core::upgrade,
    identity,
    noise,
    tcp,
    yamux,
    swarm::{NetworkBehaviour, SwarmEvent, SwarmBuilder},
    Multiaddr, PeerId, Transport,

use std::error::Error;
use std::time::Duration;

// 基础传输层配置: TCP + Noise 加密 + Yamux 流复用
fn build_transport() -> libp2p::core::transport::Boxed<
    (PeerId, libp2p::core::muxing::StreamMuxerBox),
> {
    // 1. 生成身份密钥对
    let id_keys = identity::Keypair::generate_ed25519();
    let local_peer_id = PeerId::from(id_keys.public());
    println!("本地节点 ID: {}", local_peer_id);

    // 2. 配置 Noise 加密 (类似 TLS 但为 P2P 设计)
    let noise_keys = noise::Keypair::<noise::X25519Spec>::new()
        .into_authentic(&id_keys)
        .expect("签名噪声密钥对失败");

    // 3. 构建传输层: TCP -> Noise 升级 -> Yamux 多路复用
    let transport = tcp::tokio::Transport::new(tcp::Config::default().nodelay(true))
        .upgrade(upgrade::Version::V1)
        .authenticate(noise::Config::new(noise_keys).expect("Noise 配置失败"))
        .multiplex(yamux::Config::default())
        .timeout(Duration::from_secs(20))
        .boxed();

    transport
}
```

### 1.5 实战：构建一个简单的 P2P 聊天应用

```rust
// P2P 聊天应用 —— 用 libp2p 实现两台节点互相发消息

use libp2p::{
    core::upgrade,
    gossipsub::{self, Gossipsub, GossipsubEvent, MessageAuthenticity},
    identity, noise,
    swarm::{NetworkBehaviour, SwarmEvent, SwarmBuilder},
    tcp, yamux, PeerId, Transport,
};
use tokio::io::{self, AsyncBufReadExt};
use tracing_subscriber::EnvFilter;

// ---- 定义网络行为 (NetworkBehaviour) ----
// NetworkBehaviour 是 libp2p 的核心抽象:
// 你把你需要的所有协议组合在一个 struct 中

#[derive(NetworkBehaviour)]
struct ChatBehaviour {
    // GossipSub: 发布/订阅消息广播
    gossipsub: gossipsub::Behaviour,
    // 可选: 节点识别 (让其他节点知道我们的信息)
    // identify: identify::Behaviour,
    // 可选: Kademlia (后续会用到)
    // kademlia: kad::Behaviour<kad::store::MemoryStore>,
}

// ---- 构建 Swarm ----
async fn build_chat_swarm() -> Result<
    libp2p::Swarm<ChatBehaviour>,
    Box<dyn std::error::Error>,
> {
    // 1. 生成身份
    let local_key = identity::Keypair::generate_ed25519();
    let local_peer_id = PeerId::from(local_key.public());
    println!("聊天节点 ID: {}", local_peer_id);

    // 2. 配置 GossipSub
    //    创建 gossipsub 消息的 Topic (频道)
    let gossipsub_config = gossipsub::Config::default();
    let mut gossipsub: gossipsub::Behaviour = gossipsub::Behaviour::new(
        MessageAuthenticity::Signed(local_key.clone()),
        gossipsub_config,
    )?;

    // 订阅聊天频道
    let chat_topic = gossipsub::IdentTopic::new("rust-learn-chat");
    gossipsub.subscribe(&chat_topic)?;
    println!("已订阅频道: rust-learn-chat");

    // 3. 构建传输层
    let noise_keys = noise::Keypair::<noise::X25519Spec>::new()
        .into_authentic(&local_key)?;

    let transport = tcp::tokio::Transport::new(tcp::Config::default().nodelay(true))
        .upgrade(upgrade::Version::V1)
        .authenticate(noise::Config::new(noise_keys)?)
        .multiplex(yamux::Config::default())
        .boxed();

    // 4. 创建 Swarm
    let behaviour = ChatBehaviour { gossipsub };
    let swarm = SwarmBuilder::with_tokio_executor(transport, behaviour, local_peer_id)
        .build();

    Ok(swarm)
}

// ---- 事件处理循环 ----
async fn run_chat_event_loop(
    mut swarm: libp2p::Swarm<ChatBehaviour>,
) -> Result<(), Box<dyn std::error::Error>> {
    // 监听随机端口
    swarm.listen_on("/ip4/0.0.0.0/tcp/0".parse()?)?;

    // 启动消息读取任务 (从 stdin 读用户输入)
    let mut stdin = io::BufReader::new(io::stdin()).lines();

    let chat_topic = gossipsub::IdentTopic::new("rust-learn-chat");

    loop {
        tokio::select! {
            // 处理 Swarm 事件
            event = swarm.select_next_some() => {
                match event {
                    SwarmEvent::Behaviour(ChatBehaviourEvent::Gossipsub(
                        GossipsubEvent::Message { propagation_source, message_id, message }
                    )) => {
                        // 收到消息
                        let content = String::from_utf8_lossy(&message.data);
                        println!(
                            "\n[收到] 来自 {} (消息ID: {}): {}",
                            propagation_source,
                            message_id,
                            content,
                        );
                        print!("> ");
                    }
                    SwarmEvent::NewListenAddr { address, .. } => {
                        println!("\n[网络] 监听地址: {}", address);
                        print!("> ");
                    }
                    SwarmEvent::ConnectionEstablished { peer_id, .. } => {
                        println!("\n[网络] 新连接: {}", peer_id);
                        print!("> ");
                    }
                    SwarmEvent::ConnectionClosed { peer_id, cause, .. } => {
                        println!("\n[网络] 连接断开: {} ({:?})", peer_id, cause);
                        print!("> ");
                    }
                    _ => {}
                }
            }

            // 处理用户输入
            line = stdin.next_line() => {
                if let Ok(Some(input)) = line {
                    if input.is_empty() { continue; }

                    // 特殊命令
                    if input.starts_with("/connect ") {
                        let addr: libp2p::Multiaddr = input[9..].parse()?;
                        swarm.dial(addr)?;
                        println!("正在连接到: {}", input[9..].trim());
                    } else if input == "/peers" {
                        println!("已连接节点: {} 个", swarm.connected_peers().count());
                        for peer_id in swarm.connected_peers() {
                            println!("  - {}", peer_id);
                        }
                    } else {
                        // 发送消息到频道
                        if let Err(e) = swarm.behaviour_mut().gossipsub.publish(
                            chat_topic.clone(),
                            input.as_bytes().to_vec(),
                        ) {
                            println!("发送失败: {:?}", e);
                        } else {
                            println!("[发送] {}", input);
                        }
                    }
                    print!("> ");
                }
            }
        }
    }
}

// 启动聊天应用 (两个终端运行)
// 终端A: cargo run -- --listen  # 启动第一个节点
// 终端B: cargo run -- --connect /ip4/127.0.0.1/tcp/<port>/p2p/<peerid>  # 连接到第一个节点

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    tracing_subscriber::fmt()
        .with_env_filter(EnvFilter::from_default_env())
        .init();

    let swarm = build_chat_swarm().await?;
    run_chat_event_loop(swarm).await?;

    Ok(())
}
```

## 二、Kademlia DHT：去中心化节点发现

### 2.1 Kademlia 核心概念

Kademlia 是以太坊 discv5、IPFS 和 BitTorrent 使用的分布式哈希表算法。

```rust
// Kademlia DHT 的关键概念实现 (简化版本)

use std::collections::{BTreeMap, BinaryHeap};
use std::cmp::Ordering;
use sha2::{Sha256, Digest};

// ---- XOR 距离度量 ----
// Kademlia 使用 XOR 作为节点间距离的度量

fn xor_distance(a: &[u8; 32], b: &[u8; 32]) -> [u8; 32] {
    let mut result = [0u8; 32];
    for i in 0..32 {
        result[i] = a[i] ^ b[i];
    }
    result
}

// 计算 common prefix length (CPL): 决定 k-bucket 索引
fn common_prefix_length(a: &[u8; 32], b: &[u8; 32]) -> usize {
    let dist = xor_distance(a, b);
    let mut cpl = 0;
    for byte in &dist {
        let leading_zeros = byte.leading_zeros() as usize;
        cpl += leading_zeros;
        if leading_zeros < 8 {
            break;  // 找到第一个1位
        }
    }
    cpl
}

// ---- K-Bucket: 路由表的基本单元 ----
struct KBucket {
    k: usize,                   // 每个桶最大条目数 (通常 k=20)
    peers: BTreeMap<[u8; 32], PeerInfo>,  // 按最后一次看到时间排序
    last_updated: std::time::Instant,
}

#[derive(Clone, Debug)]
struct PeerInfo {
    node_id: [u8; 32],
    ip: std::net::SocketAddr,
}

impl KBucket {
    fn new(k: usize) -> Self {
        KBucket {
            k,
            peers: BTreeMap::new(),
            last_updated: std::time::Instant::now(),
        }
    }

    fn insert(&mut self, node_id: [u8; 32], peer: PeerInfo) -> bool {
        // 如果节点已存在，移到末尾 (最近使用)
        if self.peers.contains_key(&node_id) {
            self.peers.remove(&node_id);
            self.peers.insert(node_id, peer);
            return true;
        }

        // 如果桶未满，直接插入
        if self.peers.len() < self.k {
            self.peers.insert(node_id, peer);
            return true;
        }

        // 桶已满: ping 最早插入的节点
        // 在真实实现中，如果最早节点不响应则替换它
        // 这里简化: 拒绝新节点
        false
    }
}

impl BinaryHeap<KBucket> {
    // ... 实现见下方路由表
}

// ---- 路由表: 256个 k-bucket ----
struct RoutingTable {
    local_id: [u8; 32],
    buckets: Vec<KBucket>,  // 256个桶, 每个对应一个距离前缀
    k: usize,
}

impl RoutingTable {
    fn new(local_id: [u8; 32], k: usize) -> Self {
        RoutingTable {
            local_id,
            buckets: (0..256).map(|_| KBucket::new(k)).collect(),
            k,
        }
    }

    // 确定一个节点应该放在哪个桶
    fn bucket_index(&self, node_id: &[u8; 32]) -> usize {
        let cpl = common_prefix_length(&self.local_id, node_id);
        // CPL = i 的节点放入桶 i
        cpl   // 0 到 255
    }

    // 添加节点到路由表
    fn add_node(&mut self, node_id: [u8; 32], peer: PeerInfo) -> bool {
        let idx = self.bucket_index(&node_id);
        self.buckets[idx].insert(node_id, peer)
    }

    // 查找离目标最近的 k 个节点
    fn find_closest(&self, target: &[u8; 32]) -> Vec<PeerInfo> {
        let mut candidates = BinaryHeap::new();

        // 收集所有节点
        for bucket in &self.buckets {
            for (node_id, peer) in &bucket.peers {
                let distance = xor_distance(target, node_id);
                candidates.push(DistancePeer {
                    distance,
                    peer: peer.clone(),
                });
            }
        }

        // 返回距离最近的 k 个
        let mut result = Vec::new();
        for _ in 0..self.k {
            if let Some(dp) = candidates.pop() {
                result.push(dp.peer);
            }
        }
        result
    }
}

// 用于优先队列的包装类型
#[derive(Clone)]
struct DistancePeer {
    distance: [u8; 32],
    peer: PeerInfo,
}

impl Ord for DistancePeer {
    fn cmp(&self, other: &Self) -> Ordering {
        // XOR 距离越小越优先 (所以用 reverse)
        for i in 0..32 {
            match other.distance[i].cmp(&self.distance[i]) {
                Ordering::Equal => continue,
                non_eq => return non_eq,
            }
        }
        Ordering::Equal
    }
}

impl PartialOrd for DistancePeer {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

impl PartialEq for DistancePeer {
    fn eq(&self, other: &Self) -> bool {
        self.distance == other.distance
    }
}

impl Eq for DistancePeer {}
```

### 2.2 Kademlia 的查找算法

```rust
// FIND_NODE 迭代查找

fn explain_kademlia_lookup() {
    println!("=== Kademlia 迭代查找 (FIND_NODE) ===\n");

    println!("目标: 找到离 target_id 最近的 k 个节点\n");

    println!("算法 (递归版本):");
    println!("  1. 从路由表中取离 target 最近的 α 个节点 (α=3)");
    println!("  2. 向这 α 个节点发送 FIND_NODE(target) RPC");
    println!("  3. 收到响应后，将新节点加入候选集");
    println!("  4. 从候选集中选下 α 个未询问的最近节点");
    println!("  5. 重复步骤2-4，直到没有更近的节点或已询问 k 个\n");

    println!("时间复杂度: O(log N) (网络规模 N)");
    println!("每个查询: α 个并发请求");
    println!("总跳数: ~log₂(N) ≈ 20-30 (百万级节点)");
}

// 迭代查找的伪代码实现
fn iterative_find_node(
    routing_table: &RoutingTable,
    target: &[u8; 32],
    alpha: usize,
) -> Vec<PeerInfo> {
    let k = routing_table.k;
    let mut closest = routing_table.find_closest(target);

            .cloned()
            .collect();

            break;  // 没有更多节点可以询问

        println!("第 {} 轮: 询问 {} 个节点", iteration, to_ask.len());

        for peer in &to_ask {
            asked.insert(peer.node_id);

            // closest.extend(new_nodes);
            // closest = select_k_closest(closest, target, k);
        }

        if iteration > 20 {

    }

    closest.truncate(k);
```

```rust
struct UtxoSet {
    // (txid, output_index) -> TxOutput
    utxos: HashMap<([u8; 32], u32), TxOutput>,
}

impl UtxoSet {
    fn new() -> Self {
        UtxoSet { utxos: HashMap::new() }
    }

    /// 应用一笔交易到 UTXO 集
    fn apply_transaction(&mut self, tx: &Transaction) -> Result<(), &'static str> {
        // 1. 检查所有输入是否都是未花费的
        let mut total_input = 0u64;
        for input in &tx.inputs {
            let key = (input.prev_txid, input.output_index);
            match self.utxos.remove(&key) {
                Some(output) => {
                    total_input += output.value;
                }
                None => {
                    // 恢复已删除的UTXO (回滚)
                    // 实际实现需要更完善的原子性保证
                    return Err("UTXO not found or already spent");
                }
            }
        }

        // 2. 检查输入总额 >= 输出总额 (差值 = 手续费)
        let total_output: u64 = tx.outputs.iter().map(|o| o.value).sum();
        if total_input < total_output {
            return Err("Insufficient input value");
        }

        // 3. 添加新的 UTXO
        for (i, output) in tx.outputs.iter().enumerate() {
            self.utxos.insert((tx.txid, i as u32), output.clone());
        }

        Ok(())
    }

    /// 获取地址的余额
    fn get_balance(&self, address: &[u8; 20]) -> u64 {
        self.utxos.values()
            .filter(|o| o.address == *address)
            .map(|o| o.value)
            .sum()
    }
}

// 组件3: 交易池 (Mempool)
// =============================================

struct Mempool {
    transactions: HashMap<[u8; 32], Transaction>,  // txid -> tx
    tx_order: VecDeque<[u8; 32]>,                   // 交易到达顺序
    max_size: usize,
}

impl Mempool {
    fn new(max_size: usize) -> Self {
        Mempool {
            transactions: HashMap::new(),
            tx_order: VecDeque::new(),
            max_size,
        }
    }

    fn add_transaction(&mut self, tx: Transaction) -> Result<(), &'static str> {
        // 1. 去重检查
        if self.transactions.contains_key(&tx.txid) {
            return Err("Transaction already in mempool");
        }

        // 2. 大小限制 (FIFO 驱逐)
        if self.transactions.len() >= self.max_size {
            if let Some(oldest_txid) = self.tx_order.pop_front() {
                self.transactions.remove(&oldest_txid);
            }
        }

        // 3. 添加
        self.tx_order.push_back(tx.txid);
        self.transactions.insert(tx.txid, tx);

        Ok(())
    }

    fn get_transactions(&self, max_count: usize) -> Vec<Transaction> {
        self.tx_order.iter()
            .take(max_count)
            .filter_map(|txid| self.transactions.get(txid).cloned())
            .collect()
    }
}

// =============================================
// 组件4: 区块存储 (简化版, 仅存储头)
// =============================================

struct BlockStore {
    headers: Vec<BlockHeader>,               // 区块头链
    height_by_hash: HashMap<[u8; 32], u64>,  // hash -> 高度
}

impl BlockStore {
    fn new() -> Self {
        BlockStore {
            headers: Vec::new(),
            height_by_hash: HashMap::new(),
        }
    }

    fn add_header(&mut self, header: BlockHeader) -> Result<u64, &'static str> {
        // 验证工作量证明 (简化: 检查前导零)
        if !Self::check_pow(&header) {
            return Err("Invalid proof-of-work");
        }

        let height = self.headers.len() as u64;

        // 验证链式结构 (除了创世块)
        if height > 0 {
            let prev_header = &self.headers[height as usize - 1];
            if header.prev_block_hash != Self::hash_header(prev_header) {
                return Err("Broken chain");
            }
        }

        let hash = Self::hash_header(&header);
        self.height_by_hash.insert(hash, height);
        self.headers.push(header);

        Ok(height)
    }

    fn check_pow(header: &BlockHeader) -> bool {
        use sha2::{Sha256, Digest};
        let target = u32::MAX >> header.difficulty_target;
        let mut hasher = Sha256::new();
        hasher.update(&header.prev_block_hash);
        hasher.update(&header.merkle_root);
        hasher.update(&header.timestamp.to_le_bytes());
        hasher.update(&header.nonce.to_le_bytes());
        let hash = hasher.finalize();
        let hash_prefix = u32::from_be_bytes([hash[0], hash[1], hash[2], hash[3]]);
        hash_prefix <= target
    }

    fn hash_header(header: &BlockHeader) -> [u8; 32] {
        use sha2::{Sha256, Digest};
        let mut hasher = Sha256::new();
        hasher.update(&header.prev_block_hash);
        hasher.update(&header.merkle_root);
        hasher.update(&header.timestamp.to_le_bytes());
        hasher.update(&header.nonce.to_le_bytes());
        hasher.finalize().into()
    }

    fn get_best_height(&self) -> u64 {
        if self.headers.is_empty() { 0 } else { self.headers.len() as u64 - 1 }
    }
}
```

```rust
// 使用 rust-rocksdb 持久化区块链状态

// Cargo.toml:
// rocksdb = "0.21"

use rocksdb::{Options, DB, WriteBatch};

struct BlockchainStorage {
    db: DB,
}

impl BlockchainStorage {
    /// 打开或创建数据库
    fn open(path: &str) -> Result<Self, rocksdb::Error> {
        let mut opts = Options::default();
        opts.create_if_missing(true);
        opts.set_max_open_files(1000);
        // 区块链数据对写入性能要求高
        opts.set_write_buffer_size(64 * 1024 * 1024);  // 64MB
        opts.set_max_write_buffer_number(4);
        // 启用 Bloom filter 加速点查询
        opts.set_bloom_filter(10.0, false);

        let db = DB::open(&opts, path)?;
        Ok(BlockchainStorage { db })
    }

    // ---- 区块存储 ----
    // 键设计: block:<height> -> 序列化的 BlockHeader

    fn put_block_header(&self, height: u64, header: &[u8]) -> Result<(), rocksdb::Error> {
        let key = format!("block:{}", height);
        self.db.put(key.as_bytes(), header)
    }

    fn get_block_header(&self, height: u64) -> Result<Option<Vec<u8>>, rocksdb::Error> {
        let key = format!("block:{}", height);
        match self.db.get(key.as_bytes())? {
            Some(value) => Ok(Some(value.to_vec())),
            None => Ok(None),
        }
    }

    // ---- UTXO 存储 ----
    // 键设计: utxo:<txid_hex>:<output_index> -> 序列化的 TxOutput

    fn put_utxo(
        &self,
        txid: &[u8; 32],
        output_index: u32,
        output: &[u8],
    ) -> Result<(), rocksdb::Error> {
        let key = format!(
            "utxo:{}:{}",
            hex::encode(txid),
            output_index
        );
        self.db.put(key.as_bytes(), output)
    }

    fn delete_utxo(
        &self,
        txid: &[u8; 32],
        output_index: u32,
    ) -> Result<(), rocksdb::Error> {
        let key = format!(
            "utxo:{}:{}",
            hex::encode(txid),
            output_index
        );
        self.db.delete(key.as_bytes())
    }

    // ---- 批量写入 (原子操作) ----
    fn apply_block_atomic(
        &self,
        height: u64,
        header: &[u8],
        utxos_to_add: &[([u8; 32], u32, Vec<u8>)],
        utxos_to_remove: &[([u8; 32], u32)],
    ) -> Result<(), rocksdb::Error> {
        let mut batch = WriteBatch::default();

        // 写入区块头
        let block_key = format!("block:{}", height);
        batch.put(block_key.as_bytes(), header);

        // 写入新 UTXO
        for (txid, idx, output) in utxos_to_add {
            let utxo_key = format!("utxo:{}:{}", hex::encode(txid), idx);
            batch.put(utxo_key.as_bytes(), output);
        }

        // 删除已花费 UTXO
        for (txid, idx) in utxos_to_remove {
            let utxo_key = format!("utxo:{}:{}", hex::encode(txid), idx);
            batch.delete(utxo_key.as_bytes());
        }

        // 原子提交
        self.db.write(batch)
    }

    // ---- 迭代器: 遍历所有 UTXO ----
    fn iterate_utxos(&self) -> Result<(), rocksdb::Error> {
        let mut iter = self.db.raw_iterator();
        iter.seek(b"utxo:");  // 定位到前缀

        while iter.valid() {
            if let (Some(key), Some(value)) = (iter.key(), iter.value()) {
                // 只处理 utxo: 前缀的键
                if !key.starts_with(b"utxo:") {
                    break;  // 超出范围
                }
                // 处理 UTXO ...
                println!("UTXO: {:?} -> {} bytes", key, value.len());
            }
            iter.next();
        }

        Ok(())
    }

    // ---- 列族 (Column Family) ----
    // 更优的键空间分离方式
    fn open_with_cf(path: &str) -> Result<DB, rocksdb::Error> {
        let mut opts = Options::default();
        opts.create_if_missing(true);
        opts.create_missing_column_families(true);

        let cfs = vec!["blocks", "utxos", "transactions", "metadata"];
        DB::open_cf(&opts, path, cfs)
    }
}
```

```rust
    println!("  • 根哈希唯一标识整个状态");
    println!("  • 可生成 Merkle 证明 (某键值在状态中)\n");

    println!("MPT 节点类型:");
    println!("  ┌────────────────────────────────────┐");
    println!("  │ Branch Node: 17个子节点 (16+value) │");
    println!("  │ Extension Node: 共享前缀 + 1个子节点│");
    println!("  │ Leaf Node: 剩余路径 + 值            │");
    println!("  │ Null Node: 空节点                   │");
    println!("  └────────────────────────────────────┘\n");

    println!("MPT 在以太坊中的用途:");
    println!("  1. State Trie: 所有账户的状态");
    println!("  2. Storage Trie: 每个合约的存储");
    println!("  3. Transaction Trie: 每个区块的交易");
    println!("  4. Receipt Trie: 每个区块的收据\n");

    println!("Rust 实现选项:");
    println!("  • reth-trie: Reth 客户端的 MPT 实现");
    println!("  • ethereum-trie-rs: 独立实现");
```

# 10.11 回测系统架构与实现

> **[Track B: 量化金融]** 本章从零构建一个事件驱动的 Rust 回测引擎，涵盖架构设计、策略抽象、模拟引擎、风险指标和并行参数优化。

---

## 1. 回测系统的核心问题

回测是量化交易的生命线——在投入真实资金之前，验证策略在历史数据上的表现。一个好的回测系统需要回答三个问题：

1. **策略有效吗？** — 在足够长的历史数据上产生正期望收益
2. **结果可信吗？** — 避免过拟合、幸存者偏差、未来函数泄露
3. **实盘可重现吗？** — 模拟的延迟、滑点、手续费与实际一致

```
┌─────────────────────────────────────────────────────────┐
│              回测系统的四大陷阱                            │
├─────────────────────────────────────────────────────────┤
│                                                           │
│  1. 未来函数泄露 (Look-ahead Bias)                        │
│     → 在 t 时刻使用了 t+1 的信息                           │
│     → Rust 解：事件时间戳严格单调递增断言                  │
│                                                           │
│  2. 幸存者偏差 (Survivorship Bias)                        │
│     → 历史成分股列表中只有活到最后的公司                   │
│     → Rust 解：使用 Point-in-Time 历史成分股数据库         │
│                                                           │
│  3. 过拟合 (Overfitting)                                  │
│     → 在历史数据上完美，在样本外表现差                     │
│     → Rust 解：参数分样本内/样本外，rayon 并行扫描         │
│                                                           │
│  4. 忽略交易成本 (Ignoring Costs)                         │
│     → 忽略手续费、滑点、冲击成本                           │
│     → Rust 解：在模拟引擎中显式建模所有成本                │
│                                                           │
└─────────────────────────────────────────────────────────┘
```

## 3. 回溯引擎核心 Trait

### 3.1 五大组件 Trait

```rust
// backtest_traits.rs —— 回测引擎组件 trait

/// 时钟接口 — 控制回测的时间推进
pub trait Clock: Send + Sync {
    /// 当前模拟时间
    fn now(&self) -> DateTime<Utc>;
    /// 推进到下一个时间点，返回是否还有数据
    fn advance(&mut self) -> bool;
    /// 设置回放速度倍率（1.0 = 实时, 0.0 = 最快, 2.0 = 2倍速）
    fn set_speed(&mut self, multiplier: f64);
    /// 获取已经过的事件数
    fn events_processed(&self) -> u64;
}

/// 市场数据源 — 提供历史行情数据
pub trait MarketDataFeed: Send + Sync {
    /// 获取下一个市场数据事件
    /// 返回 None 表示回测结束
    fn next_event(&mut self) -> Option<MarketEvent>;
    /// 订阅指定股票的数据
    fn subscribe(&mut self, symbols: &[String]);
    /// 获取数据源的开始/结束时间范围
    fn range(&self) -> (DateTime<Utc>, DateTime<Utc>);
    /// 获取数据源中的全部股票列表
    fn available_symbols(&self) -> Vec<String>;
}

/// 订单网关 — 将订单转换为成交（模拟撮合）
pub trait OrderGateway: Send + Sync {
    /// 提交订单，返回成交事件列表
    /// 可能产生 0 个或多个 FillEvent（市价单部分成交等）
    fn submit_order(&mut self, order: &OrderEvent) -> Vec<FillEvent>;
    /// 取消指定订单
    fn cancel_order(&mut self, order_id: &str) -> Result<(), String>;
    /// 获取当前未成交订单
    fn pending_orders(&self) -> Vec<&OrderEvent>;
    /// 更新市场状态（每次行情更新时调用）
    fn update_market(&mut self, event: &MarketEvent);
}

/// 持仓管理 — 跟踪资金、持仓、盈亏
pub trait Portfolio: Send + Sync {
    /// 处理成交事件，更新持仓
    fn on_fill(&mut self, fill: &FillEvent);
    /// 更新市场估值（Mark-to-Market）
    fn mark_to_market(&mut self, prices: &BTreeMap<String, Decimal>);
    /// 当前总权益（现金 + 持仓市值）
    fn equity(&self) -> Decimal;
    /// 当前现金余额
    fn cash(&self) -> Decimal;
    /// 获取指定股票的持仓
    fn position(&self, symbol: &str) -> Option<&Position>;
    /// 获取净值的完整时间序列
    fn equity_curve(&self) -> &[(DateTime<Utc>, Decimal)];
    /// 当前日期
    fn date(&self) -> DateTime<Utc>;
}

/// 策略接口 — 用户实现的核心逻辑
pub trait Strategy: Send + Sync {
    /// 处理市场数据事件（行情到来时调用）
    fn on_tick(&mut self, event: &MarketEvent, portfolio: &dyn Portfolio) -> Vec<SignalEvent>;
    /// 处理成交事件（订单成交后调用）
    fn on_fill(&mut self, fill: &FillEvent, portfolio: &dyn Portfolio);
    /// 策略名称
    fn name(&self) -> &str;
    /// 策略参数（用于参数优化）
    fn parameters(&self) -> BTreeMap<String, Decimal>;
    /// 用新参数重新初始化策略
    fn set_parameters(&mut self, params: &BTreeMap<String, Decimal>);
}

/// 持仓数据
#[derive(Debug, Clone)]
pub struct Position {
    pub symbol: String,
    pub quantity: Decimal,      // 正数做多，负数做空
    pub average_price: Decimal, // 平均成本
    pub market_value: Decimal,  // 当前市值
    pub unrealized_pnl: Decimal, // 浮动盈亏
    pub realized_pnl: Decimal,  // 已实现盈亏
}
```

### 3.2 回测引擎主循环

```rust
// backtest_engine.rs —— 回测引擎主循环
use std::time::Instant;

/// 回测引擎配置
pub struct BacktestConfig {
    /// 初始资金
    pub initial_capital: Decimal,
    /// 回放速度倍率（None 表示全速）
    pub replay_speed: Option<f64>,
    /// 是否启用延迟模拟
    pub enable_latency_simulation: bool,
    /// 是否启用滑点模拟
    pub enable_slippage: bool,
    /// 手续费率（bps, basis points）
    pub commission_bps: Decimal,
}

/// 回测引擎 — 协调所有组件的事件循环
pub struct BacktestEngine<C, M, G, P, S>
where
    C: Clock,
    M: MarketDataFeed,
    G: OrderGateway,
    P: Portfolio,
    S: Strategy,
{
    clock: C,
    data_feed: M,
    gateway: G,
    portfolio: P,
    strategy: S,
    config: BacktestConfig,
    /// 事件队列（按时间戳排序）
    event_queue: Vec<Box<dyn Event>>,
    /// 回测统计
    stats: BacktestStats,
}

/// 事件 trait（支持同一队列混合不同类型）
trait Event: Debug + Send {
    fn timestamp(&self) -> DateTime<Utc>;
    fn event_type(&self) -> &str;
}

impl Event for MarketEvent {
    fn timestamp(&self) -> DateTime<Utc> { self.timestamp }
    fn event_type(&self) -> &str { "MARKET" }
}

impl Event for FillEvent {
    fn timestamp(&self) -> DateTime<Utc> { self.timestamp }
    fn event_type(&self) -> &str { "FILL" }
}

impl Event for SignalEvent {
    fn timestamp(&self) -> DateTime<Utc> { self.timestamp }
    fn event_type(&self) -> &str { "SIGNAL" }
}

impl<C, M, G, P, S> BacktestEngine<C, M, G, P, S>
where
    C: Clock,
    M: MarketDataFeed,
    G: OrderGateway,
    P: Portfolio,
    S: Strategy,
{
    /// 运行回测主循环
    pub fn run(&mut self) -> Result<BacktestResult, BacktestError> {
        let start_wall = Instant::now();

        while self.clock.advance() {
            // ─── 步骤 1: 获取市场数据 ───
            if let Some(market_event) = self.data_feed.next_event() {
                // 更新 Gateway 的市场状态（用于模拟成交）
                self.gateway.update_market(&market_event);

                // ─── 步骤 2: 策略处理行情 ───
                let signals = self.strategy.on_tick(&market_event, &self.portfolio);

                for signal in &signals {
                    // ─── 步骤 3: 信号转订单 ───
                    let orders = self.signal_to_orders(signal);

                    for order in &orders {
                        // ─── 步骤 4: 提交订单给撮合网关 ───
                        let fills = self.gateway.submit_order(order);

                        for fill in &fills {
                            // ─── 步骤 5: 更新持仓 ───
                            self.portfolio.on_fill(fill);
                            // ─── 步骤 6: 通知策略成交结果 ───
                            self.strategy.on_fill(fill, &self.portfolio);
                            // ─── 步骤 7: 记录统计 ───
                            self.stats.record_fill(fill);
                        }
                    }
                }

                // ─── 步骤 8: 逐日估值（仅在日末执行）───
                // 生产系统会检查 market_event.timestamp 是否为日末
            }

            // ─── 步骤 9: 速度控制 ───
            if let Some(speed) = self.config.replay_speed {
                if speed > 0.0 {
                    // 根据虚拟时间和实际时间计算需要的 sleep
                    // std::thread::sleep(...)
                }
            }
        }

        let elapsed_wall = start_wall.elapsed();

        // ─── 汇总结果 ───
        let equity_curve = self.portfolio.equity_curve().to_vec();
        let metrics = self.calculate_metrics(&equity_curve);

        Ok(BacktestResult {
            equity_curve,
            metrics,
            stats: self.stats.clone(),
            wall_time: elapsed_wall,
        })
    }

    /// 将信号转换为订单列表
    fn signal_to_orders(&self, signal: &SignalEvent) -> Vec<OrderEvent> {
        match signal.signal_type {
            SignalType::Entry => {
                // 根据信号强度计算目标持仓量和订单
                let current_pos = self.portfolio.position(&signal.symbol);
                let target_qty = calculate_target_quantity(
                    signal.target_weight,
                    signal.strength,
                    self.portfolio.equity(),
                );
                // 生成入场订单
                vec![OrderEvent {
                    timestamp: signal.timestamp,
                    symbol: signal.symbol.clone(),
                    order_type: OrderType::Market,
                    side: if signal.strength > Decimal::ZERO { Side::Buy } else { Side::Sell },
                    quantity: target_qty.abs(),
                    limit_price: None,
                    order_id: generate_order_id(),
                }]
            }
            SignalType::Exit => {
                // 出场：平掉全部持仓
                if let Some(pos) = self.portfolio.position(&signal.symbol) {
                    vec![OrderEvent {
                        timestamp: signal.timestamp,
                        symbol: signal.symbol.clone(),
                        order_type: OrderType::Market,
                        side: if pos.quantity > Decimal::ZERO { Side::Sell } else { Side::Buy },
                        quantity: pos.quantity.abs(),
                        limit_price: None,
                        order_id: generate_order_id(),
                    }]
                } else {
                    vec![]
                }
            }
            SignalType::Adjust => {
                // 调整仓位到目标权重
                // 计算 delta 并生成对应的买卖订单
                self.generate_adjust_orders(signal)
            }
        }
    }

}

fn calculate_target_quantity(
    target_weight: Decimal,
    strength: Decimal,
    equity: Decimal,
) -> Decimal {
    equity * target_weight * strength
```

### 4.1 可配置速度回放

```rust
// market_replay.rs —— 市场数据回放引擎
use chrono::{DateTime, Duration, Utc};
use csv::ReaderBuilder;
use rust_decimal::Decimal;
use std::collections::{BTreeMap, VecDeque};
use std::path::Path;

/// CSV 市场数据回放器
pub struct CsvMarketReplay {
    /// 所有数据事件（按时间戳排序，全局有序）
    events: Vec<MarketEvent>,
    /// 当前索引位置
    cursor: usize,
    /// 数据范围
    start_time: DateTime<Utc>,
    end_time: DateTime<Utc>,
    /// 可用股票
    symbols: Vec<String>,
    /// 回放速度控制
    speed_multiplier: f64,
    /// 上次实际时间（用于速度控制）
    last_wall_time: Option<Instant>,
    /// 上次虚拟时间
    last_virtual_time: Option<DateTime<Utc>>,
}

impl CsvMarketReplay {
    /// 从 CSV 文件加载市场数据
    /// CSV 格式: timestamp,symbol,open,high,low,close,volume
    pub fn from_csv(path: impl AsRef<Path>) -> Result<Self, Box<dyn std::error::Error>> {
        let mut reader = ReaderBuilder::new()
            .has_headers(true)
            .from_path(path)?;

        let mut events = Vec::new();
        let mut symbols_set = BTreeMap::new();

        // 列索引假设（生产系统会动态检测表头）
        // timestamp,symbol,bid,ask,bid_size,ask_size,last,volume
        for result in reader.records() {
            let record = result?;
            if record.len() < 8 {
                continue;
            }

            let timestamp: DateTime<Utc> = record[0].parse()?;
            let symbol = record[1].to_string();

            let data = MarketDataType::Tick {
                bid: record[2].parse::<Decimal>()?,
                ask: record[3].parse::<Decimal>()?,
                bid_size: record[4].parse::<Decimal>()?,
                ask_size: record[5].parse::<Decimal>()?,
                last: record[6].parse::<Decimal>()?,
                volume: record[7].parse::<Decimal>()?,
            };

            symbols_set.entry(symbol.clone()).or_insert(());
            events.push(MarketEvent { timestamp, symbol, data });
        }

        // 按时间戳排序（保证正确的事件顺序）
        events.sort_by_key(|e| e.timestamp);

        let (start_time, end_time) = if events.is_empty() {
            (Utc::now(), Utc::now())
        } else {
            (events[0].timestamp, events[events.len() - 1].timestamp)
        };

        let symbols: Vec<String> = symbols_set.into_keys().collect();

        Ok(Self {
            events,
            cursor: 0,
            start_time,
            end_time,
            symbols,
            speed_multiplier: 0.0, // 默认为全速
            last_wall_time: None,
            last_virtual_time: None,
        })
    }

    /// 预加载到内存后，按时间分桶以支持多线程访问
    pub fn bucket_by_time(&self, bucket_count: usize) -> Vec<Vec<&MarketEvent>> {
        let total_ns = (self.end_time - self.start_time).num_nanoseconds().unwrap_or(1);
        let bucket_ns = total_ns / bucket_count as i64;

        let mut buckets: Vec<Vec<&MarketEvent>> = vec![vec![]; bucket_count];

        for event in &self.events {
            let offset_ns = (event.timestamp - self.start_time).num_nanoseconds().unwrap_or(0);
            let bucket_idx = (offset_ns / bucket_ns).min(bucket_count as i64 - 1);
            buckets[bucket_idx as usize].push(event);
        }

        buckets
    }
}

impl MarketDataFeed for CsvMarketReplay {
    fn next_event(&mut self) -> Option<MarketEvent> {
        if self.cursor >= self.events.len() {
            return None;
        }

        let event = self.events[self.cursor].clone();
        self.cursor += 1;

        // 速度控制：计算需要等待的时间
        if self.speed_multiplier > 0.0 {
            if let (Some(last_wall), Some(last_virtual)) =
                (self.last_wall_time, self.last_virtual_time)
            {
                let virtual_elapsed = (event.timestamp - last_virtual)
                    .num_nanoseconds()
                    .unwrap_or(0) as f64;
                let wall_should_elapse = virtual_elapsed / self.speed_multiplier;
                let wall_actually = last_wall.elapsed();
                let wall_actually_ns = wall_actually.as_nanos() as f64;

                if wall_actually_ns < wall_should_elapse {
                    let sleep_ns = (wall_should_elapse - wall_actually_ns) as u64;
                    std::thread::sleep(std::time::Duration::from_nanos(sleep_ns));
                }
            }

            self.last_wall_time = Some(Instant::now());
            self.last_virtual_time = Some(event.timestamp);
        }

        Some(event)
    }

    fn subscribe(&mut self, _symbols: &[String]) {
        // CSV 回放不涉及实时订阅
        // 如果只想测试特定股票，在运行时过滤
    }

    fn range(&self) -> (DateTime<Utc>, DateTime<Utc>) {
        (self.start_time, self.end_time)
    }

    fn available_symbols(&self) -> Vec<String> {
        self.symbols.clone()
    }
}
```

```rust
// latency_model.rs —— 延迟模拟
use chrono::{DateTime, Utc, Duration};
use rust_decimal::Decimal;
use rand::Rng;
use rand_distr::{Distribution, LogNormal};

/// 延迟模型：模拟订单传输中的网络和处理延迟
pub trait LatencyModel: Send + Sync {
    /// 计算一笔订单的模拟延迟
    fn sample_latency(&mut self, order: &OrderEvent) -> Duration;
}

/// 基于对数正态分布 + 地理距离的延迟模型
pub struct GeoDistanceLatencyModel {
    /// 基础延迟（机房内处理延迟）
    base_latency_us: f64,
    /// 地理距离（km）
    distance_km: f64,
    /// 光速在光纤中的传输速度（km/us）
    /// 光速在真空中约为 300 km/ms = 0.3 km/us
    /// 光纤中约为 200 km/ms = 0.2 km/us
    light_speed_fiber: f64,
    /// 对数正态分布的抖动
    jitter_distribution: LogNormal<f64>,
    /// 随机数生成器
    rng: rand::rngs::StdRng,
}

impl GeoDistanceLatencyModel {
    /// 创建距离模型
    /// distance_km: 距离交易所的直线距离（km）
    pub fn new(distance_km: f64, seed: u64) -> Self {
        use rand::SeedableRng;
        Self {
            base_latency_us: 10.0,       // 10 微秒基础
            distance_km,
            light_speed_fiber: 0.2,      // 光纤 ~200 km/ms = 0.2 km/us
            jitter_distribution: LogNormal::new(0.0, 0.3).unwrap(),
            rng: rand::rngs::StdRng::seed_from_u64(seed),
        }
    }
}

impl LatencyModel for GeoDistanceLatencyModel {
    fn sample_latency(&mut self, _order: &OrderEvent) -> Duration {
        // 单向传输时间（微秒）
        let one_way_us = self.base_latency_us
            + (self.distance_km / self.light_speed_fiber);

        // 添加对数正态分布的随机抖动
        let jitter: f64 = self.jitter_distribution.sample(&mut self.rng);

        // 往返时间
        let total_us = one_way_us * 2.0 * jitter;

        Duration::microseconds(total_us as i64)
    }
}

impl Clock for CsvMarketReplay {
    fn now(&self) -> DateTime<Utc> {
        if self.cursor < self.events.len() {
            self.events[self.cursor].timestamp
        } else {
            self.end_time
        }
    }

    fn advance(&mut self) -> bool {
        self.cursor < self.events.len()
    }

    fn set_speed(&mut self, multiplier: f64) {
        self.speed_multiplier = multiplier;
    }

    fn events_processed(&self) -> u64 {
        self.cursor as u64
    }
}
```

```rust
use rust_decimal::Decimal;
use rust_decimal::MathematicalOps;

/// 滑点模型
pub trait SlippageModel: Send + Sync {
    /// 计算执行价格（考虑滑点）
    /// mid_price: 当前中间价
    /// side: 买卖方向
    /// quantity: 订单数量
    /// 返回实际成交价格
    fn compute_execution_price(
        &self,
        mid_price: Decimal,
        side: Side,
        quantity: Decimal,
    ) -> Decimal;
}

/// 基于订单量相对市场深度的滑点模型
pub struct VolumeLinearSlippageModel {
    /// 每 1% 的成交量占比带来的滑点（bps）
    /// 例如：下单量为均价深度的 1%，产生 0.5 bps 滑点
    slippage_per_volume_pct_bps: Decimal,
    /// 每日平均成交量（用于归一化）
    avg_daily_volume: Decimal,
}

impl VolumeLinearSlippageModel {
    pub fn new(slippage_bps: Decimal, adv: Decimal) -> Self {
        Self {
            slippage_per_volume_pct_bps: slippage_bps,
            avg_daily_volume: adv,
        }
    }
}

impl SlippageModel for VolumeLinearSlippageModel {
    fn compute_execution_price(
        &self,
        mid_price: Decimal,
        side: Side,
        quantity: Decimal,
    ) -> Decimal {
        // 成交量占比
        let volume_pct = (quantity / self.avg_daily_volume) * Decimal::from(100);

        // 滑点（bps） = 成交量占比 * 每1%滑点
        let slippage_bps = volume_pct * self.slippage_per_volume_pct_bps;

        // 滑点金额 = 中间价 * 滑点(bps) / 10000
        let slippage_amount = mid_price * slippage_bps / Decimal::from(10000);

        match side {
            // 买入：成交价高于中间价（不利）
            Side::Buy => mid_price + slippage_amount,
            // 卖出：成交价低于中间价（不利）
            Side::Sell => mid_price - slippage_amount,
        }
    }
}

/// 手续费计算器
pub struct CommissionCalculator {
    /// 费率（bps: basis points, 万分之一）
    rate_bps: Decimal,
    /// 每笔最低手续费
    min_commission: Decimal,
}

impl CommissionCalculator {
    /// 创建手续费计算器
    /// rate_bps: 例如 1 bps = 0.01%
    pub fn new(rate_bps: Decimal, min_commission: Decimal) -> Self {
        Self { rate_bps, min_commission }
    }

    /// 计算单笔手续费
    pub fn calculate(&self, price: Decimal, quantity: Decimal) -> Decimal {
        // 成交金额
        let notional = price * quantity;
        // 手续费 = 成交金额 * 费率(bps) / 10000
        let commission = notional * self.rate_bps / Decimal::from(10000);
        // 取最大值
        commission.max(self.min_commission)
    }
}
```

### 6.1 从零实现所有指标

```rust
// backtest_metrics.rs —— 回测绩效指标计算（纯数学公式实现）
use rust_decimal::Decimal;
use rust_decimal::MathematicalOps;

/// 回测绩效指标集合
#[derive(Debug, Clone)]
pub struct BacktestResult {
    pub equity_curve: Vec<(DateTime<Utc>, Decimal)>,
    pub metrics: PerformanceMetrics,
    pub stats: BacktestStats,
    pub wall_time: std::time::Duration,
}

#[derive(Debug, Clone, Default)]
pub struct PerformanceMetrics {
    /// 年化收益率
    pub annual_return: Decimal,
    /// 年化波动率
    pub annual_volatility: Decimal,
    /// 夏普比率
    pub sharpe_ratio: Decimal,
    /// 索提诺比率（下行风险调整）
    pub sortino_ratio: Decimal,
    /// 最大回撤
    pub max_drawdown: Decimal,
    /// 最大回撤持续时间（天）
    pub max_drawdown_duration_days: i64,
    /// 卡尔玛比率（年化收益/最大回撤）
    pub calmar_ratio: Decimal,
    /// 胜率
    pub win_rate: Decimal,
    /// 盈亏比（平均盈利/平均亏损）
    pub profit_factor: Decimal,
    /// 总收益率
    pub total_return: Decimal,
    /// 交易次数
    pub total_trades: u64,
    /// 盈利交易次数
    pub winning_trades: u64,
    /// 亏损交易次数
    pub losing_trades: u64,
}

/// 从净值曲线计算回测指标
pub fn calculate_metrics(
    equity_curve: &[(DateTime<Utc>, Decimal)],
    fills: &[FillEvent],
    trading_days_per_year: i64,
    risk_free_rate: Decimal, // 如 0.04 代表 4%
) -> PerformanceMetrics {
    if equity_curve.len() < 2 {
        return PerformanceMetrics::default();
    }

    let mut metrics = PerformanceMetrics::default();
    let initial_equity = equity_curve[0].1;
    let final_equity = equity_curve[equity_curve.len() - 1].1;

    // ─── 总收益率 ───
    // total_return = (final_equity - initial_equity) / initial_equity
    metrics.total_return = (final_equity - initial_equity) / initial_equity;

    // ─── 日收益率序列 ───
    let daily_returns: Vec<Decimal> = equity_curve
        .windows(2)
        .map(|w| (w[1].1 - w[0].1) / w[0].1)
        .collect();

    // ─── 年化收益率 ───
    // annual_return = average_daily_return * trading_days
    let avg_daily_return: Decimal = daily_returns.iter().sum::<Decimal>()
        / Decimal::from(daily_returns.len());
    metrics.annual_return = avg_daily_return * Decimal::from(trading_days_per_year);

    // ─── 年化波动率 ───
    // sigma = stddev(daily_returns) * sqrt(trading_days)
    let sum_sq: Decimal = daily_returns
        .iter()
        .map(|r| (r - avg_daily_return) * (r - avg_daily_return))
        .sum();
    let daily_variance = sum_sq / Decimal::from(daily_returns.len());
    let daily_std = daily_variance.sqrt().unwrap_or(Decimal::ZERO);
    metrics.annual_volatility =
        daily_std * Decimal::from(trading_days_per_year).sqrt().unwrap();

    // ─── 夏普比率 ───
    // sharpe = (annual_return - risk_free_rate) / annual_volatility
    if metrics.annual_volatility != Decimal::ZERO {
        metrics.sharpe_ratio =
            (metrics.annual_return - Decimal::from_f64_retain(risk_free_rate.to_f64().unwrap_or(0.0)).unwrap_or(Decimal::ZERO))
                / metrics.annual_volatility;
    }

    // ─── 索提诺比率 ───
    // sortino = (annual_return - risk_free_rate) / downside_deviation_annualized
    // downside_deviation = sqrt(mean(min(r, 0)^2))
    let downside_returns: Vec<Decimal> = daily_returns
        .iter()
        .filter(|r| **r < Decimal::ZERO)
        .cloned()
        .collect();
    if !downside_returns.is_empty() {
        let sum_sq_neg: Decimal = downside_returns
            .iter()
            .map(|r| r * r)
            .sum();
        let daily_downside_variance =
            sum_sq_neg / Decimal::from(daily_returns.len()); // 注意：分母仍是总天数
        let daily_downside_std = daily_downside_variance.sqrt().unwrap_or(Decimal::ZERO);
        let annual_downside_std =
            daily_downside_std * Decimal::from(trading_days_per_year).sqrt().unwrap();
        if annual_downside_std != Decimal::ZERO {
            metrics.sortino_ratio =
                (metrics.annual_return - Decimal::from_f64_retain(risk_free_rate.to_f64().unwrap_or(0.0)).unwrap_or(Decimal::ZERO))
                    / annual_downside_std;
        }
    }

    // ─── 最大回撤 ───
    // max_drawdown = max((peak - trough) / peak)
    let mut peak = equity_curve[0].1;
    let mut max_dd = Decimal::ZERO;
    let mut dd_start: Option<DateTime<Utc>> = None;
    let mut max_dd_duration = 0i64;

    for (time, equity) in equity_curve.iter() {
        if *equity > peak {
            peak = *equity;
            dd_start = None;
        }
        let dd = (peak - equity) / peak;
        if dd > max_dd {
            max_dd = dd;
        }
        // 回撤持续时间
        if *equity < peak {
            if let Some(start) = dd_start {
                let duration_days = (*time - start).num_days();
                if duration_days > max_dd_duration {
                    max_dd_duration = duration_days;
                }
            } else {
                dd_start = Some(*time);
            }
        }
    }
    metrics.max_drawdown = max_dd;
    metrics.max_drawdown_duration_days = max_dd_duration;

    // ─── 卡尔玛比率 ───
    // calmar = annual_return / max_drawdown
    if max_dd != Decimal::ZERO {
        metrics.calmar_ratio = metrics.annual_return / max_dd;
    }

    // ─── 胜率与盈亏比（基于成交列表）───
    if !fills.is_empty() {
        // 将成交按订单号分组，计算每笔交易的盈亏
        let mut trades: BTreeMap<String, (Decimal, Decimal, Decimal)> = BTreeMap::new();
        // order_id → (buy_amount, sell_amount, total_commission)

        for fill in fills {
            let entry = trades.entry(fill.order_id.clone()).or_insert((Decimal::ZERO, Decimal::ZERO, Decimal::ZERO));
            let notional = fill.price * fill.quantity;
            match fill.side {
                Side::Buy => { entry.0 += notional; }  // 买入金额
                Side::Sell => { entry.1 += notional; } // 卖出金额
            }
            entry.2 += fill.commission;
        }

        let mut total_profit = Decimal::ZERO;
        let mut total_loss = Decimal::ZERO;
        let mut winning = 0u64;
        let mut losing = 0u64;

        for (_, (buy_amount, sell_amount, commission)) in &trades {
            // 净盈亏 = 卖出 - 买入 - 手续费
            // 对于多仓先买后卖：profit = sell - buy - commission
            // 对于空仓先卖后买：profit = sell(开仓) - buy(平仓) - commission
            let pnl = *sell_amount - *buy_amount - *commission;
            if pnl > Decimal::ZERO {
                total_profit += pnl;
                winning += 1;
            } else {
                total_loss += (-pnl); // total_loss 为正数
                losing += 1;
            }
        }

        metrics.total_trades = trades.len() as u64;
        metrics.winning_trades = winning;
        metrics.losing_trades = losing;

        if metrics.total_trades > 0 {
            metrics.win_rate = Decimal::from(winning) / Decimal::from(metrics.total_trades);
        }

        let avg_profit = if winning > 0 {
            total_profit / Decimal::from(winning)
        } else {
            Decimal::ZERO
        };
        let avg_loss = if losing > 0 {
            total_loss / Decimal::from(losing)
        } else {
            Decimal::ZERO
        };

        if avg_loss != Decimal::ZERO {
            metrics.profit_factor = avg_profit / avg_loss;
        } else if avg_profit > Decimal::ZERO {
            metrics.profit_factor = Decimal::MAX;
        }
    }

    metrics
}
```

## 7. 并行参数扫描

### 7.1 使用 rayon 并行优化

```rust
// param_sweep.rs —— 并行参数扫描
use rayon::prelude::*;
use std::collections::BTreeMap;

/// 参数扫描结果
#[derive(Debug)]
pub struct ParamSweepResult {
    /// 参数组合
    pub parameters: BTreeMap<String, Decimal>,
    /// 对应的绩效指标
    pub metrics: PerformanceMetrics,
    /// 优化目标值（如夏普比率）
    pub objective: Decimal,
}

/// 并行参数扫描引擎
pub struct ParamSweepEngine {
    /// 优化目标类型
    objective: OptimizationObjective,
    /// 优化的参数名列表
    param_names: Vec<String>,
    /// 参数搜索范围
    param_ranges: Vec<(Decimal, Decimal, Decimal)>, // (min, max, step)
    /// 并行线程数（None = rayon 自动）
    num_threads: Option<usize>,
}

#[derive(Debug, Clone, Copy)]
pub enum OptimizationObjective {
    /// 最大化夏普比率
    MaximizeSharpe,
    /// 最大化卡尔玛比率
    MaximizeCalmar,
    /// 最大化索提诺比率
    MaximizeSortino,
    /// 最小化最大回撤
    MinimizeMaxDrawdown,
    /// 最大化收益/回撤比
    MaximizeReturnDrawdown,
}

impl ParamSweepEngine {
    /// 生成所有参数组合（笛卡尔积）
    fn generate_param_combinations(&self) -> Vec<BTreeMap<String, Decimal>> {
        // 递归/迭代生成参数网格
        // 对于参数 [p1 in [0.1, 0.5]], [p2 in [10, 50]]
        // 生成: [{p1:0.1, p2:10}, {p1:0.1, p2:20}, ..., {p1:0.5, p2:50}]
        let mut combinations = vec![BTreeMap::new()];

        for (i, param_name) in self.param_names.iter().enumerate() {
            let (min, max, step) = &self.param_ranges[i];
            let mut new_combinations = vec![];

            let mut val = *min;
            while val <= *max {
                for combo in &combinations {
                    let mut new_combo = combo.clone();
                    new_combo.insert(param_name.clone(), val);
                    new_combinations.push(new_combo);
                }
                val += *step;
            }

            combinations = new_combinations;
        }

        combinations
    }

    /// 使用 rayon 并行扫描
    pub fn run_parallel<S, F>(
        &self,
        create_strategy: F, // 闭包：(params) → S
        data_feed_builder: &(dyn Fn() -> Box<dyn MarketDataFeed> + Sync),
    ) -> Vec<ParamSweepResult>
    where
        S: Strategy + Send,
        F: Fn(&BTreeMap<String, Decimal>) -> S + Send + Sync,
    {
        let combinations = self.generate_param_combinations();
        println!(
            "参数扫描: {} 个组合，{} 个并行任务",
            combinations.len(),
            rayon::current_num_threads()
        );

        let results: Vec<ParamSweepResult> = combinations
            .par_iter() // rayon 并行迭代器
            .map(|params| {
                let mut strategy = create_strategy(params);
                let mut data_feed = data_feed_builder();
                let mut portfolio = SimplePortfolio::new(Decimal::from(1_000_000)); // 100万初始资金
                let mut gateway = NaiveGateway::new();
                let clock = NullClock; // 全速回测用

                let mut engine = BacktestEngine {
                    clock,
                    data_feed,
                    gateway,
                    portfolio,
                    strategy,
                    config: BacktestConfig::default(),
                    event_queue: vec![],
                    stats: BacktestStats::default(),
                };

                let result = engine.run().unwrap_or_else(|e| {
                    eprintln!("参数扫描错误: {:?}", e);
                    BacktestResult::default()
                });

                // 提取优化目标值
                let objective = match self.objective {
                    OptimizationObjective::MaximizeSharpe => result.metrics.sharpe_ratio,
                    OptimizationObjective::MaximizeCalmar => result.metrics.calmar_ratio,
                    OptimizationObjective::MaximizeSortino => result.metrics.sortino_ratio,
                    OptimizationObjective::MinimizeMaxDrawdown => -result.metrics.max_drawdown,
                    OptimizationObjective::MaximizeReturnDrawdown => {
                        if result.metrics.max_drawdown != Decimal::ZERO {
                            result.metrics.total_return / result.metrics.max_drawdown
                        } else {
                            Decimal::ZERO
                        }
                    }
                };

                ParamSweepResult {
                    parameters: params.clone(),
                    metrics: result.metrics,
                    objective,
                }
            })
            .collect();

        // 按优化目标排序
        let mut sorted = results;
        sorted.sort_by(|a, b| {
            b.objective.partial_cmp(&a.objective).unwrap_or(std::cmp::Ordering::Equal)
        });

        sorted
    }
}
```

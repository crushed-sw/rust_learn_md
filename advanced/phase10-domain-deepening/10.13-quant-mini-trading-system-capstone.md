```rust
// channels.rs —— 系统级通道初始化
use crossbeam::queue::ArrayQueue;
use std::sync::Arc;
use tokio::sync::broadcast;

/// 系统级通道集合
/// 集中管理所有组件间的通信通道
pub struct SystemChannels {
    // ── 市场数据广播 ──
    /// 市场数据生产者端
    pub market_data_tx: broadcast::Sender<MarketDataMessage>,
    /// 市场数据消费者端（订单簿将从中订阅）
    pub market_data_rx: broadcast::Receiver<MarketDataMessage>,

    // ── 订单簿 → 策略 SPSC ──
    /// 订单簿快照队列（容量 256——约 2-5ms 的缓冲）
    pub order_book_tx: Arc<ArrayQueue<OrderBookSnapshot>>,
    pub order_book_rx: Arc<ArrayQueue<OrderBookSnapshot>>,

    // ── 策略 → 网关 SPSC ──
    /// 交易信号队列
    pub signal_tx: Arc<ArrayQueue<TradeSignal>>,
    pub signal_rx: Arc<ArrayQueue<TradeSignal>>,

    // ── 网关 → 报告 SPSC ──
    /// 执行报告队列（从网关回传给系统监控）
    pub execution_report_tx: Arc<ArrayQueue<ExecutionReportInternal>>,
    pub execution_report_rx: Arc<ArrayQueue<ExecutionReportInternal>>,

    // ── 请求-响应（同步查询）──
    /// 用于查询当前持仓/状态的 oneshot 对
    pub query_tx: tokio::sync::mpsc::Sender<SystemQuery>,
    pub query_rx: tokio::sync::mpsc::Receiver<SystemQuery>,
}

/// 市场数据消息（通过广播传输）
#[derive(Debug, Clone)]
pub struct MarketDataMessage {
    pub timestamp_ns: u64,
    pub symbol: String,
    /// 完整订单簿快照 或 Tick 增量
    pub data: MarketDataPayload,
}

#[derive(Debug, Clone)]
pub enum MarketDataPayload {
    /// L2 订单簿快照（5 档）
    L2Snapshot {
        bids: [(i64, u32); 5],  // (价格, 数量) × 5 档
        asks: [(i64, u32); 5],
    },
    /// Tick 增量（最新成交）
    Tick {
        price: i64,
        quantity: u32,
        side: u8, // 0=Bid, 1=Ask
    },
}

/// 订单簿快照（从订单簿发送给策略）
#[derive(Debug, Clone, Copy)]
pub struct OrderBookSnapshot {
    pub timestamp_ns: u64,
    pub symbol_id: u32,
    pub mid_price: i64,
    pub spread: u32,
    pub best_bid: i64,
    pub best_ask: i64,
    pub bid_volume: u64,     // 买盘深度 (前5档)
    pub ask_volume: u64,     // 卖盘深度 (前5档)
    pub vwap: i64,           // 成交量加权均价
    pub last_trade_price: i64,
    pub last_trade_qty: u32,
}

/// 交易信号（策略引擎输出）
#[derive(Debug, Clone, Copy)]
pub struct TradeSignal {
    pub timestamp_ns: u64,
    pub symbol_id: u32,
    pub signal_type: SignalType,
    pub order_type: OrderType,
    pub side: Side,
    pub quantity: u32,
    pub limit_price: i64,
    pub stop_price: i64,
    pub signal_strength: u8, // 0-100
}

/// 内部执行报告（网关内部格式）
#[derive(Debug, Clone)]
pub struct ExecutionReportInternal {
    pub timestamp_ns: u64,
    pub order_id: u64,
    pub exec_type: ExecType,
    pub ord_status: OrdStatus,
    pub symbol_id: u32,
    pub side: Side,
    pub last_qty: u32,
    pub last_price: i64,
    pub leaves_qty: u32,
    pub cum_qty: u32,
    pub avg_price: i64,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum ExecType {
    New = 0,
    PartialFill = 1,
    Fill = 2,
    Canceled = 4,
    Rejected = 8,
    Expired = 12,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum OrdStatus {
    New = 0,
    PartiallyFilled = 1,
    Filled = 2,
    Canceled = 4,
    Rejected = 8,
}

/// 系统查询请求
pub enum SystemQuery {

impl SystemChannels {
    /// 初始化所有通道
    pub fn new() -> Self {
        // 市场数据广播（容量 4096 = 约 10-40ms 缓冲）
        let (market_data_tx, market_data_rx) = broadcast::channel(4096);

        // 订单簿快照 SPSC

        // 交易信号 SPSC

        Self {
            market_data_tx,
            market_data_rx,
            order_book_tx: ob_queue.clone(),
            order_book_rx: ob_queue,
            signal_tx: signal_queue.clone(),
            signal_rx: signal_queue,
            execution_report_tx: exec_queue.clone(),
            execution_report_rx: exec_queue,
            query_tx,
            query_rx,
        }
    }
}
```

## 3. 均值回复策略实现

### 3.1 策略核心逻辑

```rust
// strategy_mean_reversion.rs —— 均值回复策略
use rust_decimal::Decimal;

/// 均值回复策略配置
pub struct MeanReversionConfig {
    /// 回望窗口（计算均值的 Tick 数量）
    pub lookback_ticks: usize,
    /// 入场阈值（价格偏离均值的倍数，以标准差为单位）
    pub entry_zscore: f64,
    /// 出场阈值（回归到多少标准差时平仓）
    pub exit_zscore: f64,
    /// 最大持仓量
    pub max_position_qty: u32,
    /// 每笔订单最大数量
    pub max_order_qty: u32,
    /// 止损比例（亏损超过此比例强制平仓）
    pub stop_loss_pct: f64,
    /// 止盈比例
    pub take_profit_pct: f64,
}

impl Default for MeanReversionConfig {
    fn default() -> Self {
        Self {
            lookback_ticks: 100,
            entry_zscore: 2.0,      // 偏离 2 个标准差时入场
            exit_zscore: 0.5,       // 回归到 0.5 标准差时出场
            max_position_qty: 1000,
            max_order_qty: 500,
            stop_loss_pct: 0.02,    // 2% 止损
            take_profit_pct: 0.05,  // 5% 止盈
        }
    }
}

/// 均值回复策略状态机
pub struct MeanReversionStrategy {
    config: MeanReversionConfig,
    /// 价格历史（环形缓冲区）
    price_history: Vec<i64>,
    /// 当前在 history 中的写入位置
    history_cursor: usize,
    /// 当前持仓量（正数多头，负数空头）
    current_position: i64,
    /// 持仓均价
    entry_price: i64,
    /// 当前状态
    state: StrategyState,
    /// 累计已实现盈亏
    total_realized_pnl: i64,
    /// 总交易次数
    total_trades: u64,
    /// 最后处理的快照时间戳（防止重复处理）
    last_snapshot_ts: u64,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum StrategyState {
    /// 等待入场信号
    Waiting,
    /// 已入场（持有多头仓位）
    LongPosition,
    /// 已入场（持有空头仓位）
    ShortPosition,
}

impl MeanReversionStrategy {
    pub fn new(config: MeanReversionConfig) -> Self {
        Self {
            price_history: vec![0; config.lookback_ticks],
            history_cursor: 0,
            current_position: 0,
            entry_price: 0,
            state: StrategyState::Waiting,
            total_realized_pnl: 0,
            total_trades: 0,
            last_snapshot_ts: 0,
            config,
        }
    }

    /// 处理每个订单簿快照——策略的核心
    /// 返回需要执行的交易信号
    pub fn on_order_book_snapshot(
        &mut self,
        snapshot: &OrderBookSnapshot,
    ) -> Option<TradeSignal> {
        // 防止重复处理同一快照
        if snapshot.timestamp_ns <= self.last_snapshot_ts {
            return None;
        }
        self.last_snapshot_ts = snapshot.timestamp_ns;

        // ── 步骤 1: 添加价格到历史 ──
        let mid_price = snapshot.mid_price;
        self.price_history[self.history_cursor] = mid_price;
        self.history_cursor = (self.history_cursor + 1) % self.config.lookback_ticks;

        // ── 步骤 2: 计算均值和标准差 ──
        let (mean, std_dev) = self.compute_mean_and_std();

        // 如果标准差太小（无波动），不交易
        if std_dev < 10 {
            return None;
        }

        // ── 步骤 3: 计算 Z-Score ──
        // z_score = (current_price - mean) / std_dev
        let z_score = (mid_price - mean) as f64 / std_dev as f64;

        // ── 步骤 4: 状态机逻辑 ──
        match self.state {
            StrategyState::Waiting => {
                // 等待入场信号
                if z_score > self.config.entry_zscore {
                    // 价格显著高于均值 → 做空
                    self.state = StrategyState::ShortPosition;
                    self.entry_price = mid_price;
                    let qty = self.calculate_entry_qty(mid_price);
                    self.current_position = -(qty as i64); // 空头为负

                    return Some(TradeSignal {
                        timestamp_ns: snapshot.timestamp_ns,
                        symbol_id: snapshot.symbol_id,
                        signal_type: SignalType::Entry,
                        order_type: OrderType::Limit,
                        side: Side::Sell,
                        quantity: qty,
                        limit_price: snapshot.best_bid, // 卖出使用最优买价
                        stop_price: 0,
                        signal_strength: ((z_score / self.config.entry_zscore * 100.0) as u8).min(100),
                    });
                } else if z_score < -self.config.entry_zscore {
                    // 价格显著低于均值 → 做多
                    self.state = StrategyState::LongPosition;
                    self.entry_price = mid_price;
                    let qty = self.calculate_entry_qty(mid_price);
                    self.current_position = qty as i64;

                    return Some(TradeSignal {
                        timestamp_ns: snapshot.timestamp_ns,
                        symbol_id: snapshot.symbol_id,
                        signal_type: SignalType::Entry,
                        order_type: OrderType::Limit,
                        side: Side::Buy,
                        quantity: qty,
                        limit_price: snapshot.best_ask, // 买入使用最优卖价
                        stop_price: 0,
                        signal_strength: ((-z_score / self.config.entry_zscore * 100.0) as u8).min(100),
                    });
                }
                None // 无信号
            }
            StrategyState::LongPosition => {
                // 持仓中——检查出场条件
                let pnl_pct = (mid_price - self.entry_price) as f64 / self.entry_price as f64;

                // 条件 1: 价格回归到均值附近 → 止盈出场
                if z_score.abs() < self.config.exit_zscore {
                    return self.generate_exit_signal(snapshot, pnl_pct);
                }
                // 条件 2: 止损
                if pnl_pct < -self.config.stop_loss_pct {
                    tracing::warn!("触发止损: PnL={:.4}%", pnl_pct * 100.0);
                    return self.generate_exit_signal(snapshot, pnl_pct);
                }
                // 条件 3: 止盈
                if pnl_pct > self.config.take_profit_pct {
                    tracing::info!("触发止盈: PnL={:.4}%", pnl_pct * 100.0);
                    return self.generate_exit_signal(snapshot, pnl_pct);
                }

                None
            }
            StrategyState::ShortPosition => {
                let pnl_pct = (self.entry_price - mid_price) as f64 / self.entry_price as f64;

                // 条件 1: 价格回归到均值附近 → 止盈出场
                if z_score.abs() < self.config.exit_zscore {
                    return self.generate_exit_signal(snapshot, pnl_pct);
                }
                // 条件 2: 止损
                if pnl_pct < -self.config.stop_loss_pct {
                    return self.generate_exit_signal(snapshot, pnl_pct);
                }
                // 条件 3: 止盈
                if pnl_pct > self.config.take_profit_pct {
                    return self.generate_exit_signal(snapshot, pnl_pct);
                }

                None
            }
        }
    }

    /// 生成出场交易信号
    fn generate_exit_signal(
        &mut self,
        snapshot: &OrderBookSnapshot,
        pnl_pct: f64,
    ) -> Option<TradeSignal> {
        let qty = self.current_position.unsigned_abs() as u32;
        let (side, limit_price) = if self.current_position > 0 {
            (Side::Sell, snapshot.best_bid)
        } else {
            (Side::Buy, snapshot.best_ask)
        };

        // 记录已实现盈亏
        let pnl_ticks = (snapshot.mid_price - self.entry_price) * self.current_position;
        self.total_realized_pnl += pnl_ticks;
        self.total_trades += 1;

        println!(
            "出场: PnL={:.4}%, ticks={}, 总PnL={}",
            pnl_pct * 100.0, pnl_ticks, self.total_realized_pnl
        );

        // 重置状态
        self.state = StrategyState::Waiting;
        self.current_position = 0;
        self.entry_price = 0;

        Some(TradeSignal {
            timestamp_ns: snapshot.timestamp_ns,
            symbol_id: snapshot.symbol_id,
            signal_type: SignalType::Exit,
            order_type: OrderType::Market, // 出场用市价单
            side,
            quantity: qty,
            limit_price,
            stop_price: 0,
            signal_strength: 100,
        })
    }

    /// 计算买/卖入场量（基于波动率调整）
    fn calculate_entry_qty(&self, _mid_price: i64) -> u32 {
        // 简单实现：固定数量（生产系统会根据波动率动态调整）
        self.config.max_order_qty.min(self.config.max_position_qty)
    }

    /// 计算滑动窗口内的均值与标准差
    fn compute_mean_and_std(&self) -> (i64, i64) {
        let n = self.price_history.len() as f64;
        let sum: i64 = self.price_history.iter().sum();
        let mean = (sum as f64 / n) as i64;

        let sum_sq_diff: f64 = self.price_history
            .iter()
            .map(|&p| {
                let diff = p as f64 - mean as f64;
                diff * diff
            })
            .sum();
        let variance = sum_sq_diff / n;
        let std_dev = variance.sqrt() as i64;

        (mean, std_dev)
    }
}
```

```rust
pub struct SimExchange {

    pub fn new(config: SimExchangeConfig) -> Self {

            mid_prices: BTreeMap::new(),

    /// 启动模拟交易所（tokio 任务）
    pub async fn run(
        config: SimExchangeConfig,
        shutdown_rx: tokio::sync::watch::Receiver<bool>,
    ) {
        let listener = TcpListener::bind(&config.bind_addr).await.unwrap();
        println!("模拟交易所启动: {}", config.bind_addr);

        let exchange = Arc::new(tokio::sync::Mutex::new(SimExchange::new(config)));

        loop {
            tokio::select! {
                accept_result = listener.accept() => {
                    match accept_result {
                        Ok((stream, addr)) => {
                            println!("模拟交易所: 新连接来自 {}", addr);
                            let ex = exchange.clone();
                            let mut shutdown = shutdown_rx.clone();
                            tokio::spawn(async move {
                                SimExchange::handle_connection(ex, stream, shutdown).await;
                            });
                        }
                        Err(e) => {
                            eprintln!("接受连接失败: {}", e);
                        }
                    }
                }
                _ = shutdown_rx.changed() => {
                    println!("模拟交易所收到停止信号");
                    break;
                }
            }
        }
    }

    /// 处理一个 FIX 连接
    async fn handle_connection(
        exchange: Arc<tokio::sync::Mutex<SimExchange>>,
        mut stream: tokio::net::TcpStream,
        shutdown: tokio::sync::watch::Receiver<bool>,
    ) {
        let mut buf = vec![0u8; 4096];

        loop {
            tokio::select! {
                read_result = stream.read(&mut buf) => {
                    match read_result {
                        Ok(0) => {
                            println!("模拟交易所: 客户端断开连接");
                            break;
                        }
                        Ok(n) => {
                            // 解析 FIX 消息
                            let msg_str = String::from_utf8_lossy(&buf[..n]);
                            println!("模拟交易所 收到: {}", msg_str.trim());

                            // 模拟处理延迟
                            let latency_us = {
                                let ex = exchange.lock().await;
                                ex.config.latency_us
                                    + rand::random::<u64>() % ex.config.latency_jitter_us
                            };
                            tokio::time::sleep(
                                std::time::Duration::from_micros(latency_us)
                            ).await;

                            // 处理订单
                            let response = {
                                let mut ex = exchange.lock().await;
                                ex.process_fix_message(&msg_str)
                            };

                            // 发送响应
                            if let Some(resp) = response {
                                if let Err(e) = stream.write_all(resp.as_bytes()).await {
                                    eprintln!("模拟交易所发送失败: {}", e);
                                    break;
                                }
                                println!("模拟交易所 发送: {}", resp.trim());
                            }
                        }
                        Err(e) => {
                            eprintln!("模拟交易所读取错误: {}", e);
                            break;
                        }
                    }
                }
                _ = shutdown.changed() => {
                    break;
                }
            }
        }
    }

    /// 处理 FIX 消息——模拟撮合
    fn process_fix_message(&mut self, msg: &str) -> Option<String> {
        // 简化版 FIX 解析（生产环境用完整的 FIX parser）
        let msg_type = extract_fix_tag(msg, "35");
        match msg_type.as_str() {
            "D" => self.handle_new_order(msg),   // NewOrderSingle
            "F" => self.handle_cancel(msg),       // OrderCancelRequest
            "5" => self.handle_logout(msg),       // Logout
            _ => None,
        }
    }

    /// 处理新订单——撮合逻辑
    fn handle_new_order(&mut self, msg: &str) -> Option<String> {
        let symbol = extract_fix_tag(msg, "55");
        let side = extract_fix_tag(msg, "54"); // 1=Buy, 2=Sell
        let price: i64 = extract_fix_tag(msg, "44").parse().unwrap_or(0);
        let qty: u32 = extract_fix_tag(msg, "38").parse().unwrap_or(0);

        // 简单撮合：随机部分成交
        let fill_qty = if rand::random::<f64>() < 0.8 {
            qty  // 80% 概率全量成交
        } else {
            qty * (rand::random::<u32>() % 50 + 50) / 100  // 50-99% 部分成交
        };

        let is_partial = fill_qty < qty;
        let exec_type = if is_partial { "1" } else { "2" }; // 1=PartialFill, 2=Fill
        let ord_status = if is_partial { "1" } else { "2" };

        // 构建 ExecutionReport
        let order_id = extract_fix_tag(msg, "11");
        let exec_report = format!(
            "8=FIX.4.4\x019=150\x0135=8\x0149=EXCHANGE\x0156=CLIENT\x01\
             34=1\x0152=20260705-10:00:00.000\x0111={}\x0117={}\x01\
             150={}\x0139={}\x0155={}\x0154={}\x0131={}\x0132={}\x01\
             14={}\x016={}\x01151={}\x0110=000\x01",
            order_id,  // 11: ClOrdID
            "EXEC001", // 17: ExecID
            exec_type, // 150: ExecType
            ord_status, // 39: OrdStatus
            symbol,    // 55: Symbol
            side,      // 54: Side
            price,     // 31: LastPx
            fill_qty,  // 32: LastQty
            fill_qty,  // 14: CumQty
            price,     // 6: AvgPx
            qty - fill_qty, // 151: LeavesQty
        );

        Some(exec_report)
    }

    fn handle_cancel(&self, msg: &str) -> Option<String> {
        let order_id = extract_fix_tag(msg, "11");
        // 模拟取消确认
        Some(format!(
            "8=FIX.4.4\x019=100\x0135=8\x0149=EXCHANGE\x0156=CLIENT\x01\
             34=1\x0152=20260705-10:00:00.000\x0111={}\x0117=CANCEL001\x01\
             150=4\x0139=4\x0110=000\x01",
            order_id
        ))
    }

    fn handle_logout(&self, _msg: &str) -> Option<String> {
        Some("8=FIX.4.4\x019=60\x0135=5\x0149=EXCHANGE\x0156=CLIENT\x01\
              34=1\x0152=20260705-10:00:00.000\x0158=Logout Accepted\x0110=000\x01"
            .to_string())
    }
}

/// 从 FIX 消息中提取 tag 值（简化版）
fn extract_fix_tag(msg: &str, tag: &str) -> String {
    let search_start = format!("{}=", tag);
    if let Some(pos) = msg.find(&search_start) {
        let start = pos + search_start.len();
        let rest = &msg[start..];
        if let Some(end) = rest.find('\x01') {
            return rest[..end].to_string();
        }
        return rest.to_string();
    }
    String::new()
}
```

### 8.3 musl 静态链接

```rust
// musl_compile.sh —— 使用 musl 进行完全静态链接
//
// # 添加 musl target
// rustup target add x86_64-unknown-linux-musl
//
// # 安装 musl-tools
// apt-get install musl-tools
//
// # 编译静态链接二进制
// RUSTFLAGS="-C target-feature=+crt-static -C link-self-contained=yes" \
//   cargo build --release --target x86_64-unknown-linux-musl
//
// # 验证静态链接（应该输出 "not a dynamic executable"）
// ldd target/x86_64-unknown-linux-musl/release/mini-trading-system
//
// musl 静态链接的优点:
// 1. 无运行时依赖 (无需 libc.so 版本匹配)
// 2. Docker 镜像可基于 scratch (空镜像，~5MB vs ~100MB)
// 3. 跨 Linux 发行版兼容 (glibc vs musl 无冲突)
// 4. 部署更简单 (单文件复制即可运行)
//
// musl 的缺点:
// 1. musl 的 malloc 在多线程下性能不如 jemalloc
//    → 使用 #[global_allocator] 替换为 jemalloc
// 2. DNS 解析行为与 glibc 不同 (musl 逐个查询 vs glibc 并行)
//    → 使用 trust-dns-resolver 代替 libc 的 getaddrinfo
```

### 8.4 物理机 vs 虚拟机

```
┌─────────────────────────────────────────────────────────────┐
│           部署环境对比: 物理机 vs 虚拟机                      │
├─────────────────┬───────────────────┬───────────────────────┤
│ 维度            │ 物理机 (Bare Metal)│ 虚拟机 (VM/云实例)    │
├─────────────────┼───────────────────┼───────────────────────┤
│ CPU 独占        │ 100% 控制 (isolcpus)│ 虚拟化层额外延迟     │
│ 内存延迟        │ ~60-100 ns        │ ~80-150 ns (+20-50%)  │
│ 网络延迟        │ < 1 us (DPDK)     │ ~5-20 us (虚拟NIC)   │
│ NUMA 控制       │ 精确 pin          │ 受 hypervisor 限制    │
│ TSC 稳定性      │ 恒定              │ 可能有漂移            │
│ 中断确定性      │ 可控 (IRQ affinity)│ 不可控 (hypervisor)  │
│ 部署灵活性      │ 低 (硬件锁定)     │ 高 (快照/迁移/弹性)  │
│ 推荐场景        │ 超低延迟 (HFT)    │ 中低频率策略/回测     │
├─────────────────┼───────────────────┼───────────────────────┤
│ 建议:                                                         │
│ - 对延迟要求 < 10us → 物理机 + DPDK                          │
│ - 对延迟要求 < 100us → 虚拟机 + io_uring + SR-IOV            │
│ - 回测/研究 → 任意环境，优先用 Docker 容器                   │
└──────────────────────────────────────────────────────────────┘
```

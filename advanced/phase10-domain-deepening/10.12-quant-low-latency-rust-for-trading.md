## 1. 延迟预算分析

### 1.1 延迟从哪里来？

在低延迟交易中，每一微秒都至关重要。理解延迟的根本组成是优化的第一步。

```
┌─────────────────────────────────────────────────────────────────┐
│                   交易系统端到端延迟预算                          │
├─────────────────────────────────────────────────────────────────┤
│                                                                   │
│  总计: ~5-20 us（微秒）                                           │
│                                                                   │
│  ┌─────────┬─────────┬─────────┬──────────┬────────────────┐    │
│  │ 网络传输 │ 内核协议 │ 应用解析 │ 策略计算  │ 订单构建+发送 │    │
│  │ 1-10 us │ 0.5-5 us│ 0.1-2 us│ 0.05-1 us│  0.1-1 us     │    │
│  └─────────┴─────────┴─────────┴──────────┴────────────────┘    │
│     ▲           ▲          ▲          ▲            ▲              │
│     │           │          │          │            │              │
│  DPDK缩短     io_uring/  SIMD解析   分支预测   预分配buffer      │

│  └───────────────────────────────────────────────────────────┘  │
│                                                                   │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 延迟度量基准

```rust
// latency_baseline.rs —— 建立延迟基准
use std::arch::x86_64::_rdtsc;
use std::time::Instant;

/// 使用 RDTSC 获取处理器时间戳计数（最精确的墙上时间）
/// 注意：在多 socket 系统上，TSC 可能不同步
#[cfg(target_arch = "x86_64")]
pub fn rdtsc() -> u64 {
    unsafe { _rdtsc() }
}

/// 将 TSC 差值转换为纳秒（需要校准频率）
pub struct TscCalibration {
    /// TSC ticks per nanosecond
    ticks_per_ns: f64,
}

impl TscCalibration {
    /// 通过 sleep 校准 TSC 频率
    /// 在程序启动时执行一次
    pub fn calibrate() -> Self {
        let sleep_duration = std::time::Duration::from_millis(100);

        let tsc_start = rdtsc();
        std::thread::sleep(sleep_duration);
        let tsc_end = rdtsc();

        let tsc_diff = tsc_end.wrapping_sub(tsc_start);
        let ns = sleep_duration.as_nanos() as f64;
        let ticks_per_ns = tsc_diff as f64 / ns;

        println!("TSC 校准完成: {:.2} ticks/ns", ticks_per_ns);
        Self { ticks_per_ns }
    }

    /// 将 RDTSC 差值转为纳秒
    pub fn tsc_to_ns(&self, tsc_diff: u64) -> f64 {
        tsc_diff as f64 / self.ticks_per_ns
    }
}

/// 延迟度量宏——测量代码块执行时间
/// 使用 RDTSC 以亚纳秒分辨率度量
#[macro_export]
macro_rules! measure_latency {
    ($calibration:expr, $label:expr, $block:block) => {{
        let tsc_before = $crate::rdtsc();
        let result = $block;
        let tsc_after = $crate::rdtsc();
        let ns = $calibration.tsc_to_ns(tsc_after.wrapping_sub(tsc_before));
        println!("[{}] 耗时: {:.1} ns", $label, ns);
        result
    }};
}
```

## 2. 消除堆分配

### 2.1 Arena 分配器

在交易系统的热路径中，`Box::new()` 或 `Vec::push()` 的行为是致命的——每次堆分配都需要进入内核获取内存。

```rust
// arena_trading.rs —— 交易系统 Arena 分配器
use bumpalo::Bump;
use std::cell::UnsafeCell;

/// 线程局部的订单对象 Arena
/// 每个策略线程拥有一个独立的 Arena，
/// 所有订单对象从 Arena 分配，批量释放。
///
/// 原理：Arena 维护一个大的预分配内存块，
/// 通过简单的 bump-pointer 分配（仅移动指针），
/// 时间复杂度 O(1)，无系统调用。
pub struct OrderArena {
    /// bumpalo 内部维护了多个内存块
    bump: UnsafeCell<Bump>,
}

impl OrderArena {
    /// 创建 Arena（预分配 1 MB）
    pub fn new() -> Self {
        Self {
            bump: UnsafeCell::new(Bump::with_capacity(1024 * 1024)),
        }
    }

    /// 从 Arena 分配一个订单对象
    /// 返回 'arena 生命周期的引用
    pub fn alloc_order<'a>(&'a self, order: NewOrder) -> &'a mut NewOrder {
        let bump = unsafe { &mut *self.bump.get() };
        bump.alloc(order)
    }

    /// 处理完一批消息后重置 Arena
    /// 所有之前分配的对象变为无效
    pub fn reset(&self) {
        let bump = unsafe { &mut *self.bump.get() };
        bump.reset();
    }
}

/// 低延迟订单结构（无堆分配）
/// 所有字段在栈上或 Arena 中
#[derive(Debug, Clone)]
pub struct NewOrder {
    pub symbol: [u8; 16],    // 固定大小字符数组 vs String
    pub price: i64,           // 定点数：价格 * 10000
    pub quantity: u32,
    pub side: OrderSide,
    pub order_type: OrderType,
    pub order_id: u64,
    pub flags: u8,            // 位标志（IOC, FOK, etc.）
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
#[repr(u8)]
pub enum OrderSide {
    Buy = 0,
    Sell = 1,
}

/// 静态预分配最大订单数
/// 比 Arena 更激进：完全消除运行时分配
const MAX_OUTSTANDING_ORDERS: usize = 10000;

pub struct StaticOrderPool {
    /// 预分配的订单数组（编译时确定大小）
    orders: Box<[UnsafeCell<NewOrder>; MAX_OUTSTANDING_ORDERS]>,
    /// 空闲槽位栈（用 Vec 但可以替换为固定数组）
    free_slots: UnsafeCell<Vec<usize>>,
}

impl StaticOrderPool {
    /// 初始化预分配池
    pub fn new() -> Self {
        let orders = unsafe {
            // 用 MaybeUninit 构建未初始化数组（避免 Default 开销）
            use std::mem::MaybeUninit;
            let mut boxed: Box<[MaybeUninit<UnsafeCell<NewOrder>>; MAX_OUTSTANDING_ORDERS]> =
                Box::new(unsafe { MaybeUninit::uninit().assume_init() });
            // 初始化每个元素
            for i in 0..MAX_OUTSTANDING_ORDERS {
                boxed[i] = MaybeUninit::new(UnsafeCell::new(NewOrder {
                    symbol: [0u8; 16],
                    price: 0,
                    quantity: 0,
                    side: OrderSide::Buy,
                    order_type: OrderType::Limit,
                    order_id: 0,
                    flags: 0,
                }));
            }
            // 类型转换：从 MaybeUninit 转为已初始化
            std::mem::transmute::<
                Box<[MaybeUninit<UnsafeCell<NewOrder>>; MAX_OUTSTANDING_ORDERS]>,
                Box<[UnsafeCell<NewOrder>; MAX_OUTSTANDING_ORDERS]>,
            >(boxed)
        };

        let mut free_slots = Vec::with_capacity(MAX_OUTSTANDING_ORDERS);
        for i in (0..MAX_OUTSTANDING_ORDERS).rev() {
            free_slots.push(i);
        }

        Self {
            orders,
            free_slots: UnsafeCell::new(free_slots),
        }
    }

    /// 获取一个空闲的订单槽位索引
    pub fn alloc(&self) -> Option<usize> {
        let free = unsafe { &mut *self.free_slots.get() };
        free.pop()
    }

    /// 释放一个订单槽位
    pub fn free(&self, index: usize) {
        let free = unsafe { &mut *self.free_slots.get() };
        free.push(index);
    }

    /// 获取订单的不可变引用
    pub fn get(&self, index: usize) -> &NewOrder {
        unsafe { &*self.orders[index].get() }
    }

    /// 获取订单的可变引用
    pub fn get_mut(&self, index: usize) -> &mut NewOrder {
        unsafe { &mut *self.orders[index].get() }
    }
}
```

### 2.2 smallvec：栈上小数组

```rust
// smallvec_usage.rs —— 使用 smallvec 避免小型容器的堆分配
use smallvec::{SmallVec, smallvec};

/// 一笔订单的部分成交列表
/// 大多数订单只产生 0-2 次成交，少量大订单有 10+ 次
/// SmallVec<[T; 4]>: 前 4 个元素在栈上，超出才堆分配
type FillList = SmallVec<[FillEntry; 4]>;

#[derive(Debug, Clone)]
pub struct FillEntry {
    pub price: i64,
    pub quantity: u32,
    pub timestamp_ns: u64,
}

/// 订单成交拆分逻辑
pub fn split_order_into_fills(
    order: &NewOrder,
    market_depth: &[(i64, u32)],
) -> FillList {
    let mut fills: FillList = smallvec![]; // 栈上分配
    let mut remaining = order.quantity;

    for (level, &(price, avail_qty)) in market_depth.iter().enumerate() {
        if remaining == 0 { break; }

        let fill_qty = remaining.min(avail_qty);
        fills.push(FillEntry {
            price,
            quantity: fill_qty,
            timestamp_ns: 0, // 模拟中填充
        });

        remaining -= fill_qty;
    }

    fills
}
```

## 3. 缓存行优化

### 3.1 缓存行基础

现代 x86 CPU 的缓存行是 64 字节。当两个线程修改同一缓存行内的不同字段时，发生 **False Sharing**——即使它们操作的是互不相关的数据。

```
┌─────────────────────────────────────────────────────────────┐
│               False Sharing 问题                              │
├─────────────────────────────────────────────────────────────┤
│                                                               │
│  缓存行 0 (64 bytes)                                          │
│  ┌───────────────┬───────────────┬─────────────────────────┐ │
│  │ Core 0 写入   │ Core 1 写入   │                         │ │
│  │ field_a (4B)  │ field_b (4B)  │     (未使用)            │ │
│  └───────────────┴───────────────┴─────────────────────────┘ │
│                                                               │
│  问题: Core 0 写 field_a → 缓存行在 Core 1 无效化             │
│        Core 1 写 field_b → 缓存行在 Core 0 无效化             │
│        结果: 缓存行在两个 Core 间乒乓，性能暴跌 10-100x       │
│                                                               │
│  解决: 每个热点字段占用独立的缓存行                            │
│  ┌────────────────┬────────────────┬───────────────────────┐ │
│  │ field_a (4B)   │ (60B padding) │      缓存行 0          │ │
│  ├────────────────┼────────────────┼───────────────────────┤ │
│  │ field_b (4B)   │ (60B padding) │      缓存行 1          │ │
│  └────────────────┴────────────────┴───────────────────────┘ │
│                                                               │
└─────────────────────────────────────────────────────────────┘
```

## 4. 分支预测优化

### 4.1 likely/unlikely 提示

```rust
// branch_prediction.rs —— 分支预测优化
use std::intrinsics::{likely, unlikely};

/// 价格比较——对所有价格档位做分支无关比较
/// 避免分支预测失败导致的流水线停顿（~15-20 周期）

/// 分支无关的取最大值（不会产生分支指令）
#[inline(always)]
pub fn branchless_max(a: i64, b: i64) -> i64 {
    // x - ((x - y) & ((x - y) >> 63))
    // 在 x86 上编译为 cmov (conditional move) 指令
    let diff = a.wrapping_sub(b);
    a.wrapping_sub(diff & (diff >> 63))
}

/// 分支无关的取最小值
#[inline(always)]
pub fn branchless_min(a: i64, b: i64) -> i64 {
    let diff = b.wrapping_sub(a);
    a.wrapping_add(diff & (diff >> 63))
}

/// 订单簿最佳价格比较——热路径
/// 例如：比较所有潜在的成交价，找到最优买价
#[inline(always)]
pub fn best_bid_price(bids: &[i64]) -> i64 {
    // 使用分支无关的 max 累计比较
    // 注意：这在小数组（< 10 档）上比分支版本更快
    // 在大数组上，带分支的循环可能更好（分支预测器有足够信息）
    let mut best = i64::MIN;
    for &bid in bids {
        best = branchless_max(best, bid);
    }
    best
}

/// 限价单撮合——热路径的分支优化
/// 判断订单价格是否满足成交条件
#[inline(always)]
pub fn can_fill(order_side: OrderSide, order_price: i64, market_price: i64) -> bool {
    // 传统写法（有分支）:
    // if order_side == OrderSide::Buy {
    //     order_price >= market_price  // 买单价 >= 卖一价 → 成交
    // } else {
    //     order_price <= market_price  // 卖单价 <= 买一价 → 成交
    // }

    // 分支无关写法:
    // 将 bool 转为 i64 (true→-1, false→0)
    // 然后使用位运算选择比较结果
    let is_buy = (order_side as u8 == 0) as i64; // 1 = buy, 0 = sell
    // 对 buy:  order_price >= market_price → order_price - market_price >= 0 → diff >= 0
    // 对 sell: order_price <= market_price → market_price - order_price >= 0 → -diff >= 0
    let diff = order_price.wrapping_sub(market_price);
    let buy_fillable = (diff >= 0) as i64;
    let sell_fillable = (diff <= 0) as i64;
    // 根据 side 选择
    let result = (is_buy & buy_fillable) | ((1 - is_buy) & sell_fillable);
    result != 0
}

/// 使用 likely/unlikely 提示编译器
/// 标记大概率/小概率分支
pub fn handle_order_type(order_type: OrderType, price: i64, qty: u32) -> i64 {
    // 90% 的订单是限价单 → likely
    if unsafe { likely(order_type == OrderType::Limit) } {
        // 限价单逻辑（热路径，放在分支预测大概率侧）
        price * qty as i64
    }
    // 5% 市价单 → unlikely
    else if unsafe { unlikely(order_type == OrderType::Market) } {
        // 市价单逻辑
        0 // 市价单没有价格
    }
    // 5% 止损单 → unlikely
    else {
        price * qty as i64 / 2 // 止损价格计算
    }
}
```

## 5. 无锁管道

### 5.1 SPSC 队列连接组件

交易系统中，市场数据 → 策略 → 订单网关 之间的数据传递是延迟的核心路径。使用无锁队列消除互斥开销。

```rust
// lockfree_pipeline.rs —— 无锁 SPSC 管道
use crossbeam::queue::ArrayQueue;
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;

/// 市场数据 → 策略引擎 的无锁通道
pub struct MarketDataChannel {
    /// 有界 SPSC 队列（环形缓冲区）
    /// ArrayQueue 使用原子操作实现，无锁无等待
    queue: ArrayQueue<PaddedMarketData>,
    /// 丢弃计数（当队列满时）
    dropped_count: AtomicU64,
    /// 最后入队时间戳
    last_enqueue_ns: AtomicU64,
}

/// 填充对齐后的市场数据（确保在队列中不 false share）
#[repr(C, align(64))]
struct PaddedMarketData {
    data: HotMarketData,
    _pad: [u8; 64 - std::mem::size_of::<HotMarketData>()],
}

impl MarketDataChannel {
    /// 创建容量为 1024 的无锁队列
    pub fn new() -> Arc<Self> {
        Arc::new(Self {
            queue: ArrayQueue::new(1024),
            dropped_count: AtomicU64::new(0),
            last_enqueue_ns: AtomicU64::new(0),
        })
    }

    /// 生产者端：推送市场数据（在数据接收线程调用）
    /// 非阻塞——如果队列满则丢弃最老的数据
    pub fn push(&self, data: HotMarketData) -> Result<(), HotMarketData> {
        let padded = PaddedMarketData {
            data,
            _pad: [0u8; 64 - std::mem::size_of::<HotMarketData>()],
        };
        match self.queue.push(padded) {
            Ok(()) => {
                self.last_enqueue_ns.store(0, Ordering::Release); // 简化
                Ok(())
            }
            Err(padded) => {
                // 队列满——丢弃数据（记录计数）
                self.dropped_count.fetch_add(1, Ordering::Relaxed);
                // 在超低延迟系统中，丢弃数据比阻塞等待更好
                Err(padded.data)
            }
        }
    }

    /// 消费者端：弹出市场数据（在策略线程调用）
    /// 非阻塞——队列空时立即返回 None
    pub fn pop(&self) -> Option<HotMarketData> {
        self.queue.pop().map(|p| p.data)
    }

    /// 批量弹出（减少原子操作开销）
    pub fn pop_batch(&self, out: &mut [HotMarketData]) -> usize {
        let mut count = 0;
        for slot in out.iter_mut() {
            match self.queue.pop() {
                Some(padded) => {
                    *slot = padded.data;
                    count += 1;
                }
                None => break,
            }
        }
        count
    }

    /// 获取被丢弃的消息数
    pub fn dropped(&self) -> u64 {
        self.dropped_count.load(Ordering::Relaxed)
    }
}

/// 策略 → 订单网关 的无锁通道
pub type SignalChannel = ArrayQueue<AlignedTradeSignal>;

/// 构建完整的交易管道
pub fn build_trading_pipeline() -> (
    Arc<MarketDataChannel>,  // 生产者端
    Arc<MarketDataChannel>,  // 消费者端
    Arc<SignalChannel>,      // 信号通道
) {
    let md_channel = MarketDataChannel::new();
    let signal_channel = Arc::new(ArrayQueue::new(256));

    (md_channel.clone(), md_channel, signal_channel)
}

/// 策略线程主循环
pub fn strategy_loop(
    md_channel: Arc<MarketDataChannel>,
    signal_channel: Arc<SignalChannel>,
) {
    let mut batch: [HotMarketData; 32] = unsafe { std::mem::zeroed() };

    loop {
        // 批量取市场数据
        let count = md_channel.pop_batch(&mut batch);

        for i in 0..count {
            let md = &batch[i];

            // ── 策略计算 ──
            // 例如：均值回复信号的快速计算
            let mid = (md.bid + md.ask) / 2;
            let signal = compute_signal(mid, md.last_price);

            if signal != 0 {
                // 推送信号
                let trade_signal = AlignedTradeSignal {
                    data: UnsafeCell::new(TradeSignalData {
                        symbol: [0u8; 8],
                        signal_price: mid,
                        signal_qty: 100,
                        signal_type: signal as u8,
                        confidence: 100,
                        timestamp_ns: md.timestamp_ns,
                    }),
                };
                let _ = signal_channel.push(trade_signal);
            }
        }
    }
}

fn compute_signal(mid: i64, last: i64) -> i32 {
    // 简化的信号计算：价格偏离均价超过阈值则产生信号
    let threshold = 5; // 5 个价格单位
    let diff = last - mid;
    if diff > threshold {
        1 // 卖出信号
    } else if diff < -threshold {
        -1 // 买入信号
    } else {
        0 // 无信号
    }
}
```

```rust
/// 生产环境的 CPU 布局建议:
///
/// ┌──────────────────────────────────────────────────────┐
/// │                CPU 核心分配方案                       │
/// ├──────────────────────────────────────────────────────┤
/// │                                                        │
/// │  Core 0:     操作系统 / 中断处理 (不要用)              │
/// │  Core 1:     网络数据接收线程 (DPDK / io_uring)       │
/// │  Core 2:     市场数据处理 + 订单簿维护                  │
/// │  Core 3-4:   策略计算 (每个策略一个核心)               │
/// │  Core 5:     订单网关 + 发送                            │
/// │  Core 6+:    日志 / 监控 / 管理接口                    │
/// │                                                        │
/// │  内核启动参数: isolcpus=1-5                            │
/// │  效果: Linux 内核不会在这些核心上调度任何进程          │
/// │         (也不会处理中断，除非手动绑定)                  │
/// │                                                        │
/// └──────────────────────────────────────────────────────┘

/// 获取隔离的核心列表（从 /proc/cmdline 解析 isolcpus）
pub fn get_isolated_cores() -> Vec<usize> {
    let cmdline = std::fs::read_to_string("/proc/cmdline").unwrap_or_default();
    for part in cmdline.split_whitespace() {
        if part.starts_with("isolcpus=") {
            let cores_str = part.trim_start_matches("isolcpus=");
            // 解析范围: "1-5,10-12" → [1,2,3,4,5,10,11,12]
            return parse_core_range(cores_str);
        }
    }
    vec![]
}

fn parse_core_range(s: &str) -> Vec<usize> {
    let mut cores = vec![];
    for part in s.split(',') {
        if let Some(pos) = part.find('-') {
            let start: usize = part[..pos].parse().unwrap_or(0);
            let end: usize = part[pos + 1..].parse().unwrap_or(0);
            cores.extend(start..=end);
        } else {
            cores.push(part.parse().unwrap_or(0));
        }
    }
    cores
}
```

```rust
/// NUMA 节点感知的内存分配
/// 在 NUMA 系统中，CPU 访问本地内存比远程快 ~1.5-2x
pub struct NumaAwareAllocator {

impl NumaAwareAllocator {
    pub fn new() -> Option<Self> {
        let topology = Topology::new().ok()?;
        Some(Self { topology })
    }

    /// 获取当前 CPU 所在的 NUMA 节点
    pub fn current_numa_node(&self) -> Option<usize> {
        let cpu_set = self.topology.get_cpubind(
            hwloc::CpuSet::which_on(hwloc::WhichCpuSet::PU, hwloc::CpuSetNoneStrategy::Unknown)
                .ok()?,
            hwloc::ObjectType::NUMANode,
            hwloc::CPUBIND_THREAD,
        );

        // 遍历 NUMA 节点，找到当前 CPU 所属的节点
        let mut depth = 0;
        for obj in self.topology.objects_at_depth(depth) {
            if obj.object_type() == ObjectType::NUMANode {
                // 检查该 NUMA 节点是否包含当前 CPU
                if obj.cpuset().is_some() {

            depth += 1;
        }
        None
    }

/// 将内存页锁定在物理内存中（防止 swap）
/// mlock 防止关键内存页被换出到磁盘
pub fn lock_memory_pages(ptr: *const u8, size: usize) {
    unsafe {
        #[cfg(target_os = "linux")]
        {
            let ret = libc::mlock(ptr as *const libc::c_void, size);
            if ret != 0 {
                eprintln!("mlock 失败: {}", std::io::Error::last_os_error());
            }
        }
    }
}

/// 预分配大页内存（Huge Pages）
/// 使用 2MB 或 1GB 大页减少 TLB 缺失
pub fn allocate_huge_pages(size: usize) -> *mut u8 {
    #[cfg(target_os = "linux")]
    {
        unsafe {
            let ptr = libc::mmap(
                std::ptr::null_mut(),
                size,
                libc::PROT_READ | libc::PROT_WRITE,
                libc::MAP_PRIVATE | libc::MAP_ANONYMOUS | libc::MAP_HUGETLB,
                -1,
                0,
            );
            if ptr == libc::MAP_FAILED {
                // 回退到普通 mmap（可能未开启大页支持）
                libc::mmap(
                    std::ptr::null_mut(),
                    size,
                    libc::PROT_READ | libc::PROT_WRITE,
                    libc::MAP_PRIVATE | libc::MAP_ANONYMOUS,
                    -1,
                    0,
                )
            } else {
                ptr
            }
        }
    }
    #[cfg(not(target_os = "linux"))]
    {
        unreachable!()
    }
}
```

```rust
fn percentile_from_sorted(sorted: &[f64], percentile: f64) -> f64 {
    if sorted.is_empty() {
        return 0.0;
    }
    let idx = (percentile / 100.0) * (sorted.len() - 1) as f64;
    let lower = sorted[idx.floor() as usize];
    let upper = sorted[idx.ceil() as usize];
    let frac = idx - idx.floor();
    lower + (upper - lower) * frac
}
```

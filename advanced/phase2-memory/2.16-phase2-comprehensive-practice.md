## 2. Demo 1: 手写智能指针全家桶 —— MyBox → MyRc → MyCell → MyCow

### 2.1 设计哲学

> "智能指针是 Rust 内存模型的名片。标准库的 `Box`、`Rc`、`Cell`、`Cow` 看起来像魔法——但它们只是普通结构体 + 几行 unsafe 代码。理解了它们的内部机制，就不再畏惧 unsafe Rust——你会知道哪部分需要 unsafe、哪部分是编译器保证的。"

本 Demo 按"所有权复杂度递增"的顺序，从零实现四个智能指针：**MyBox**（独占所有权）→ **MyRc**（共享所有权）→ **MyCell / MyRefCell**（内部可变性）→ **MyCow**（延迟克隆）。

### 2.2 `MyBox<T>` —— 独占所有权的堆指针

```rust
use std::alloc::{self, Layout};
use std::ops::{Deref, DerefMut};
use std::ptr::NonNull;

/// 手写 Box：拥有堆上分配的 T
///
/// 核心要素：
/// - NonNull<T>: 非空裸指针的零成本包装（2.13 Niche 优化）
/// - Layout: 描述类型的内存布局（2.5 Layout/Alignment）
/// - Drop: 释放堆内存（2.4 Drop）
/// - Deref: 透明访问内部值（2.3 Deref）
pub struct MyBox<T: ?Sized> {
    ptr: NonNull<T>,
}

impl<T> MyBox<T> {
    pub fn new(value: T) -> Self {
        // ━━━ ZST 特殊处理（2.9 ZST）━━━
        // ZST 不需要分配堆内存——大小是 0
        // NonNull::dangling() 返回一个按 T 对齐的"假"指针（但永远不会被解引用）
        if std::mem::size_of::<T>() == 0 {
            return MyBox {
                ptr: NonNull::dangling(),
            };
        }

        // ━━━ Layout: 描述内存需求（2.5 Layout/Alignment）━━━
        let layout = Layout::new::<T>();

        // ━━━ 堆分配 ━━━
        // SAFETY: layout 的 size > 0（ZST 已在上面处理）
        let ptr = unsafe { alloc::alloc(layout) };
        if ptr.is_null() {
            alloc::handle_alloc_error(layout);
        }

        // ━━━ MaybeUninit + ptr::write（2.10 MaybeUninit）━━━
        // 把值写入堆内存——不需要先初始化再赋值
        unsafe { ptr.cast::<T>().write(value) };

        MyBox {
            // SAFETY: ptr 非空（已检查），且正确对齐
            ptr: unsafe { NonNull::new_unchecked(ptr.cast::<T>()) },
        }
    }
}

// ━━━ Deref: 让 MyBox<T> 像 &T 一样使用（2.3 Deref）━━━
impl<T: ?Sized> Deref for MyBox<T> {
    type Target = T;

    fn deref(&self) -> &T {
        // SAFETY: ptr 始终有效（MyBox 存在 ⟹ 堆分配存在）
        unsafe { self.ptr.as_ref() }
    }
}

impl<T: ?Sized> DerefMut for MyBox<T> {
    fn deref_mut(&mut self) -> &mut T {
        unsafe { self.ptr.as_mut() }
    }
}

// ━━━ Drop: 释放堆内存（2.4 Drop）━━━
impl<T: ?Sized> Drop for MyBox<T> {
    fn drop(&mut self) {
        // ZST 没有分配堆内存——跳过释放
        if std::mem::size_of::<T>() == 0 {
            return;
        }

        let layout = Layout::for_value(unsafe { self.ptr.as_ref() });

        // SAFETY:
        // - ptr 是由 alloc::alloc 分配的
        // - layout 与分配时一致
        // - 先 drop 值，再释放内存
        unsafe {
            std::ptr::drop_in_place(self.ptr.as_ptr());
            alloc::dealloc(self.ptr.as_ptr().cast::<u8>(), layout);
        }
    }
}

// ━━━ Niche 优化验证（2.13 Niche）━━━
// MyBox<T> 使用 NonNull<T>，而非 *mut T
// 这允许 Option<MyBox<T>> 利用 null 表示 None，大小等于 MyBox<T>
fn verify_niche() {
    assert_eq!(
        std::mem::size_of::<MyBox<i32>>(),
        std::mem::size_of::<Option<MyBox<i32>>>(),
    );
    // NonNull 的 niche 优化：null 指针值表示 None
}
```

### 2.3 `MyRc<T>` —— 共享所有权的引用计数指针

```rust
use std::cell::Cell;
use std::marker::PhantomData;

/// RcInner: 堆上的控制块
///
/// 布局: { strong_count, weak_count, value }
/// 这与标准库 Rc 的内存布局一致（2.2 Rc 内部实现）
struct RcInner<T> {
    strong_count: Cell<usize>,  // Cell 允许通过共享引用修改计数（2.15 Cell）
    weak_count: Cell<usize>,
    value: T,
}

pub struct MyRc<T> {
    // 指向堆上控制块的指针
    ptr: NonNull<RcInner<T>>,
    // ━━━ PhantomData 标注所有权（2.6 PhantomData）━━━
    // MyRc<T> 拥有 T——即当 MyRc<T> 被 drop 时，T 也应该被 drop
    _marker: PhantomData<RcInner<T>>,
}

impl<T> MyRc<T> {
    pub fn new(value: T) -> Self {
        let inner = RcInner {
            strong_count: Cell::new(1),
            weak_count: Cell::new(1),  // 第一个 strong 也计入 weak_count
            value,
        };

        let layout = Layout::new::<RcInner<T>>();
        let ptr = unsafe { alloc::alloc(layout) };
        if ptr.is_null() {
            alloc::handle_alloc_error(layout);
        }
        unsafe { ptr.cast::<RcInner<T>>().write(inner) };

        MyRc {
            ptr: unsafe { NonNull::new_unchecked(ptr.cast::<RcInner<T>>()) },
            _marker: PhantomData,
        }
    }

    /// 获取强引用计数
    pub fn strong_count(this: &Self) -> usize {
        unsafe { this.ptr.as_ref() }.strong_count.get()
    }

    /// 获取弱引用计数
    pub fn weak_count(this: &Self) -> usize {
        unsafe { this.ptr.as_ref() }.weak_count.get()
    }
}

// ━━━ Clone for MyRc: 增加引用计数 ━━━
impl<T> Clone for MyRc<T> {
    fn clone(&self) -> Self {
        let inner = unsafe { self.ptr.as_ref() };
        inner.strong_count.set(inner.strong_count.get() + 1);
        MyRc {
            ptr: self.ptr,
            _marker: PhantomData,
        }
    }
}

// ━━━ Deref（2.3 Deref）━━━
impl<T> Deref for MyRc<T> {
    type Target = T;
    fn deref(&self) -> &T {
        &unsafe { self.ptr.as_ref() }.value
    }
}

// ━━━ Drop: 减少计数，最后一个释放值 ━━━
impl<T> Drop for MyRc<T> {
    fn drop(&mut self) {
        let inner = unsafe { self.ptr.as_ref() };

        // 减少 strong_count
        let old_strong = inner.strong_count.get();
        inner.strong_count.set(old_strong - 1);

        if old_strong == 1 {
            // 最后一个强引用：
            // 1. drop 值（2.4 Drop 顺序）
            unsafe { std::ptr::drop_in_place(&mut inner.ptr.as_ref().value as *const T as *mut T) };

            // 2. 减少 weak_count（因为 strong 也贡献了一个 weak）
            let old_weak = inner.weak_count.get();
            inner.weak_count.set(old_weak - 1);

            if old_weak == 1 {
                // 没有任何引用（强或弱）了——释放控制块
                let layout = Layout::new::<RcInner<T>>();
                unsafe {
                    alloc::dealloc(self.ptr.as_ptr().cast::<u8>(), layout);
                }
            }
        }
    }
}

// ━━━ MyWeak<T>: 弱引用 ━━━
pub struct MyWeak<T> {
    ptr: NonNull<RcInner<T>>,
}

impl<T> MyWeak<T> {
    /// 尝试升级为强引用
    /// 如果 strong_count == 0（值已释放），返回 None
    pub fn upgrade(&self) -> Option<MyRc<T>> {
        let inner = unsafe { self.ptr.as_ref() };
        let count = inner.strong_count.get();
        if count == 0 {
            None
        } else {
            inner.strong_count.set(count + 1);
            Some(MyRc {
                ptr: self.ptr,
                _marker: PhantomData,
            })
        }
    }
}

impl<T> Clone for MyWeak<T> {
    fn clone(&self) -> Self {
        let inner = unsafe { self.ptr.as_ref() };
        inner.weak_count.set(inner.weak_count.get() + 1);
        MyWeak { ptr: self.ptr }
    }
}

impl<T> Drop for MyWeak<T> {
    fn drop(&mut self) {
        let inner = unsafe { self.ptr.as_ref() };
        let old = inner.weak_count.get();
        inner.weak_count.set(old - 1);
        if old == 1 {
            // 最后一个 weak——释放控制块（值已经在 strong drop 时释放了）
            let layout = Layout::new::<RcInner<T>>();
            unsafe { alloc::dealloc(self.ptr.as_ptr().cast::<u8>(), layout); }
        }
    }
}

// ━━━ 型变验证（2.6 PhantomData & 型变）━━━
// MyRc<T> 对 T 是协变的：如果 'long: 'short，则 MyRc<&'long T> 可以当作 MyRc<&'short T>
// 这由 PhantomData<RcInner<T>> 自动推导（因为 RcInner<T> 拥有 T）
fn verify_variance() {
    fn _covariant<'short, 'long: 'short>(rc: MyRc<&'long str>) -> MyRc<&'short str> {
        rc  // ✓ MyRc<T> 对 T 协变——长生命周期可以缩短
    }
}
```

### 2.4 `MyCell<T> + MyRefCell<T>` —— 内部可变性实战

```rust
use std::cell::UnsafeCell;

/// ━━━ UnsafeCell: 内部可变性的唯一底层原语（2.15 UnsafeCell）━━━
///
/// UnsafeCell<T> 是编译器"认识"的特殊类型：
/// - 编译器不会对 UnsafeCell 内的值做别名优化
/// - 所有内部可变性类型最终都基于它
///
/// 注意：UnsafeCell<T> 自身是 !Sync 的（通过包含 UnsafeCell 间接实现）

/// ━━━ MyCell<T>: Copy 类型的零开销内部可变性 ━━━
///
/// 策略：整体替换（get/set），无需借用跟踪
/// 只适用于 T: Copy 类型
#[derive(Debug)]
pub struct MyCell<T> {
    inner: UnsafeCell<T>,
}

impl<T: Copy> MyCell<T> {
    pub fn new(value: T) -> Self {
        MyCell { inner: UnsafeCell::new(value) }
    }

    /// get: 返回值的拷贝——不返回引用
    /// 这就是为什么只需要 T: Copy
    pub fn get(&self) -> T {
        // SAFETY: 只读访问；调用者获得的是拷贝，不影响原始值
        unsafe { *self.inner.get() }
    }

    /// set: 整体替换——不需要知道旧值是什么
    pub fn set(&self, value: T) {
        // SAFETY: 写入一个新值；不读取旧值，不创建引用
        unsafe { *self.inner.get() = value };
    }

    /// replace: 取出旧值，放入新值
    pub fn replace(&self, value: T) -> T {
        let old = self.get();
        self.set(value);
        old
    }
}

// ━━━ MyCell 使用示例：共享计数器 ━━━
fn demo_mycell() {
    let counter = MyCell::new(0u64);
    counter.set(counter.get() + 1);
    counter.set(counter.get() + 1);
    assert_eq!(counter.get(), 2);
}

/// ━━━ MyRefCell<T>: 运行时借用检查的内部可变性 ━━━
///
/// 策略：维护 borrow 计数器，运行时检查借用规则
/// 不要求 T: Copy——返回的是真正的引用
pub struct MyRefCell<T> {
    inner: UnsafeCell<T>,
    // borrow_flag: 0 = 未借用, usize::MAX = 可变借用中, n = n 个不可变借用
    borrow_flag: Cell<isize>,
}

impl<T> MyRefCell<T> {
    pub fn new(value: T) -> Self {
        MyRefCell {
            inner: UnsafeCell::new(value),
            borrow_flag: Cell::new(0),
        }
    }

    /// 不可变借用——可以多个同时存在
    pub fn borrow(&self) -> MyRef<'_, T> {
        let flag = self.borrow_flag.get();
        if flag < 0 {
            panic!("already mutably borrowed");
        }
        self.borrow_flag.set(flag + 1);
        MyRef {
            // SAFETY: borrow_flag >= 0，没有可变借用存在
            ptr: self.inner.get(),
            cell: self as *const Self,
        }
    }

    /// 可变借用——只能有一个
    pub fn borrow_mut(&self) -> MyRefMut<'_, T> {
        if self.borrow_flag.get() != 0 {
            panic!("already borrowed");
        }
        self.borrow_flag.set(-1);
        MyRefMut {
            ptr: self.inner.get(),
            cell: self as *const Self,
        }
    }
}

// ━━━ MyRef: 不可变借用的 RAII 守卫 ━━━
pub struct MyRef<'a, T> {
    ptr: *const T,
    cell: *const MyRefCell<T>,
}

impl<T> Deref for MyRef<'_, T> {
    type Target = T;
    fn deref(&self) -> &T {
        unsafe { &*self.ptr }
    }
}

impl<T> Drop for MyRef<'_, T> {
    fn drop(&mut self) {
        let cell = unsafe { &*self.cell };
        cell.borrow_flag.set(cell.borrow_flag.get() - 1);
    }
}

// ━━━ MyRefMut: 可变借用的 RAII 守卫 ━━━
pub struct MyRefMut<'a, T> {
    ptr: *mut T,
    cell: *const MyRefCell<T>,
}

impl<T> Deref for MyRefMut<'_, T> {
    type Target = T;
    fn deref(&self) -> &T {
        unsafe { &*self.ptr }
    }
}

impl<T> DerefMut for MyRefMut<'_, T> {
    fn deref_mut(&mut self) -> &mut T {
        unsafe { &mut *self.ptr }
    }
}

impl<T> Drop for MyRefMut<'_, T> {
    fn drop(&mut self) {
        let cell = unsafe { &*self.cell };
        cell.borrow_flag.set(0);  // 释放可变借用
    }
}
```

### 2.5 MyCow<'a, B> —— 写时复制

```rust
use std::borrow::Borrow;

/// 手写 Cow：在"借用"和"拥有"之间自动切换
///
/// 核心：只在真正需要修改时才克隆（2.7 Cow）
/// 对于 90% 读 / 10% 写的场景，避免不必要的分配
pub enum MyCow<'a, B: ?Sized + 'a>
where
    B: ToOwned,
{
    Borrowed(&'a B),
    Owned(<B as ToOwned>::Owned),
}

impl<'a, B: ?Sized + 'a> MyCow<'a, B>
where
    B: ToOwned,
{
    /// 获取不可变引用——无论 Borrowed 还是 Owned
    pub fn as_ref(&self) -> &B {
        match self {
            MyCow::Borrowed(b) => b,
            MyCow::Owned(o) => o.borrow(),
        }
    }

    /// 获取可变引用——如果当前是 Borrowed，先克隆
    pub fn to_mut(&mut self) -> &mut B::Owned {
        match self {
            MyCow::Owned(o) => o,
            MyCow::Borrowed(b) => {
                // 克隆时机：真正需要修改时（2.7 Cow 的核心）
                *self = MyCow::Owned(b.to_owned());
                match self {
                    MyCow::Owned(o) => o,
                    _ => unreachable!(),
                }
            }
        }
    }
}

// ━━━ Cow 使用示例：零拷贝字符串处理 ━━━
fn demo_cow() {
    use std::borrow::Cow;

    fn process(input: Cow<'_, str>) -> Cow<'_, str> {
        if input.contains("bad") {
            // 需要修改——触发克隆
            let mut owned = input.into_owned();
            owned = owned.replace("bad", "good");
            Cow::Owned(owned)
        } else {
            // 不需要修改——零拷贝传递
            input
        }
    }

    // 场景 1：输入是借用 → 不需要修改 → 零拷贝
    let result = process(Cow::Borrowed("hello world"));
    assert!(matches!(result, Cow::Borrowed(_)));  // 仍然是 Borrowed！

    // 场景 2：输入是借用 → 需要修改 → 才克隆
    let result = process(Cow::Borrowed("this is bad code"));
    assert!(matches!(result, Cow::Owned(_)));    // 现在才是 Owned
}
```

### 2.6 智能指针抉择速查

| 需求 | 选指针 | 关键特征 |
|------|--------|----------|
| 单一所有权 + 堆分配 | `MyBox<T>` | 最简单的堆指针 |
| 共享所有权 + 不可变 | `MyRc<T>` | 引用计数，单线程 |
| 共享所有权 + 跨线程 | `Arc<T>` (标准库) | 原子引用计数 |
| 通过 `&T` 修改数据 | `MyCell<T>` / `MyRefCell<T>` | 内部可变 |
| 借用优先，偶尔拥有 | `MyCow<'a, B>` | 延迟克隆 |

### 2.7 概念映射表

## 3. Demo 2: 链表实战 —— 三种智能指针 × 两种链表

### 3.1 设计哲学

> "用 Rust 写链表是 Rust 内存模型的终极测验。不是因为它难——而是因为链表恰好踩中了 Rust 所有权的所有痛点：递归所有权、共享可变、Drop 顺序。"

### 3.2 单向链表 —— MyBox 所有权链

```rust
/// 单向链表：每个节点拥有下一个节点
///
/// 递归类型：Node<T> 包含 Option<MyBox<Node<T>>>
/// MyBox 的大小是 8 字节（指针），所以 Node<T> 的大小是确定的
enum SList<T> {
    Cons(T, MyBox<SList<T>>),
    Nil,
}

impl<T> SList<T> {
    pub fn new() -> Self {
        SList::Nil
    }

    pub fn prepend(self, value: T) -> Self {
        SList::Cons(value, MyBox::new(self))
    }

    pub fn head(&self) -> Option<&T> {
        match self {
            SList::Cons(value, _) => Some(value),
            SList::Nil => None,
        }
    }

    pub fn tail(&self) -> Option<&SList<T>> {
        match self {
            SList::Cons(_, rest) => Some(rest),
            SList::Nil => None,
        }
    }
}

// ━━━ Drop 问题：递归 Drop 可能爆栈 ━━━
//
// 默认的 Drop 顺序：
//   drop SList::Cons(v, box_tail)
//     → drop v
//     → drop box_tail
//       → drop SList::Cons(v2, box_tail2)  ← 递归！
//         → ...
//
// 对于 N 个节点的链表，这是 N 层递归 Drop——可能爆栈！
// 解决方案：手动实现迭代 Drop

impl<T> Drop for SList<T> {
    fn drop(&mut self) {
        // 迭代式 Drop：用循环代替递归
        let mut current = std::mem::replace(self, SList::Nil);
        while let SList::Cons(_, mut next) = current {
            // next 是 MyBox<SList<T>>
            // 用 replace 取出 box 内的 SList，然后手动 drop box
            current = std::mem::replace(&mut *next, SList::Nil);
            // next (MyBox) 在这里被 drop——只递归一层！
            drop(next);
        }
    }
}

// ━━━ 使用示例 ━━━
fn demo_slist() {
    let list = SList::new()
        .prepend(3)
        .prepend(2)
        .prepend(1);

    assert_eq!(list.head(), Some(&1));
    assert_eq!(list.tail().and_then(|t| t.head()), Some(&2));
}
```

### 3.3 双向链表 —— MyRc + MyRefCell 共享可变

```rust
use std::cell::RefCell;
use std::rc::Rc;

/// 双向链表节点
///
/// Rc<RefCell<Node<T>>> 的语义：
/// - Rc: 共享所有权——多个地方可以持有同一个节点
/// - RefCell: 内部可变——即使有共享引用，仍能修改 prev/next
struct DNode<T> {
    value: T,
    prev: Option<Rc<RefCell<DNode<T>>>>,
    next: Option<Rc<RefCell<DNode<T>>>>,
}

struct DList<T> {
    head: Option<Rc<RefCell<DNode<T>>>>,
    tail: Option<Rc<RefCell<DNode<T>>>>,
}

impl<T> DList<T> {
    pub fn new() -> Self {
        DList { head: None, tail: None }
    }

    /// 在尾部追加元素
    pub fn push_back(&mut self, value: T) {
        let new_node = Rc::new(RefCell::new(DNode {
            value,
            prev: self.tail.clone(),
            next: None,
        }));

        match self.tail.take() {
            Some(old_tail) => {
                // 链接旧尾部 → 新节点
                old_tail.borrow_mut().next = Some(new_node.clone());
                // 新节点的 prev 已经设置（在构造中）
                self.tail = Some(new_node);
            }
            None => {
                // 空链表：新节点既是头也是尾
                self.head = Some(new_node.clone());
                self.tail = Some(new_node);
            }
        }
    }

    /// 正向遍历
    pub fn iter_forward(&self) -> Vec<Rc<RefCell<DNode<T>>>>
    where
        T: Clone,
    {
        let mut result = Vec::new();
        let mut current = self.head.clone();
        while let Some(node) = current {
            let next = node.borrow().next.clone();
            result.push(node);
            current = next;
        }
        result
    }

    /// 获取头部值
    pub fn front(&self) -> Option<std::cell::Ref<'_, T>> {
        self.head.as_ref().map(|node| {
            // borrow() 返回的是 Ref<Node<T>>，需要手动投影到 Ref<T>
            // 这里用 unsafe 简化——标准库中没有直接的办法
            // 实际项目中用 Ref::map 或 ouroboros
            todo!("用 Ref::map 或 unsafe 实现投影")
        })
    }
}

// ━━━ 双向链表的内存布局 ━━━
//
//   head                          tail
//    │                             │
//    ▼                             ▼
//  ┌─────────┐    ┌─────────┐    ┌─────────┐
//  │ Rc ─────┼───→│ Rc ─────┼───→│ Rc      │
//  │         │←───┤         │←───┤         │
//  │ Node(1) │    │ Node(2) │    │ Node(3) │
//  └─────────┘    └─────────┘    └─────────┘
//
// 每个 Rc 的强引用计数：
// - Node(1): 1 (list.head) + 1 (Node(2).prev) = 2
// - Node(2): 1 (Node(1).next) + 1 (Node(3).prev) = 2
// - Node(3): 1 (list.tail) + 1 (Node(2).next) = 2
//
// ━━━ Drop 循环引用问题 ━━━
// 如果 list 被 drop：
// 1. drop(list.tail) → Node(3) strong_count: 2 → 1（未释放！）
// 2. drop(list.head) → Node(1) strong_count: 2 → 1（未释放！）
// 3. 但 Node(1) 和 Node(3) 通过 Node(2) 相互持有 → 内存泄漏！
//
// 解决方案：用 Weak 替代其中一方向的 Rc
```

### 3.4 Drop 深度对比：递归 vs 迭代 vs 循环引用

```rust
/// ━━━ 三种 Drop 场景对比 ━━━
///
/// 1. 单向链表 Box 递归 Drop
///    问题：N 个节点 = N 层递归 → 爆栈
///    解决：手动实现迭代 Drop
///
/// 2. 双向链表 Rc 循环引用
///    问题：head ↔ tail 相互持有 → 永远不会 Drop
///    解决：一个方向用 Weak

use std::rc::Weak;

struct SafeDNode<T> {
    value: T,
    prev: Option<Weak<RefCell<SafeDNode<T>>>>,   // ← Weak! 不增加 strong_count
    next: Option<Rc<RefCell<SafeDNode<T>>>>,     // ← Rc
}

// 当 list 被 drop：
// - head 的 strong_count 递减，可能达到 0
// - Node 的 next (Rc) 的递减沿链表传播
// - prev (Weak) 不阻止释放
// → 所有节点依次释放，没有循环引用
```

## 4. Demo 3: Pin + 自引用 —— 移动语义的"禁区"

### 4.1 设计哲学

> "Rust 的核心承诺之一：移动一个值不会产生悬垂指针——编译器通过 move 语义保证这一点。但当你构造了自引用结构（一个字段指向同结构体的另一个字段），这个承诺就被打破了。`Pin` 不是魔法——它是类型系统对'这个值不可移动'的承诺，让你在安全的边界内使用自引用。"

### 4.2 自引用解析器 —— 移动的致命后果

```rust
/// ━━━ 一个典型（但危险）的自引用结构 ━━━
///
/// buffer 拥有 JSON 数据，parsed_fields 指向 buffer 中的子串
/// 移动 SelfReferential → buffer 换了地址 → parsed_fields 悬垂！
struct SelfReferential<'a> {
    buffer: String,
    // 指向 buffer 内部数据的切片引用
    parsed_fields: Option<&'a str>,
}

impl<'a> SelfReferential<'a> {
    fn new(json: &str) -> Self {
        // 先构造——parsed_fields 还无法指向 buffer
        // （因为 buffer 还没被创建，或者说创建了但没有地址保证）
        Self {
            buffer: String::from(json),
            parsed_fields: None,
        }
    }
}

// ━━━ 问题演示：移动破坏了自引用 ━━━
fn demonstrate_move_problem() {
    // 这不安全——所以编译器不允许
    // 实际上这个结构体根本无法用安全代码正确构造
    // 因为 SelfReferential 的 parsed_fields 生命周期 'a
    // 无法与自身的 buffer 关联
}
```

### 4.3 Pin 方案 —— 保证不可移动

```rust
use std::marker::PhantomPinned;
use std::pin::Pin;

/// ━━━ Pin 安全的自引用结构 ━━━
///
/// 关键：
/// 1. PhantomPinned 使类型 !Unpin —— 编译器知道它不能安全移动
/// 2. 构造时必须用 Pin::new() 或 pin!() 宏
/// 3. 一旦构造完成，指针就稳定了
struct PinnedParser {
    buffer: String,
    // 存储偏移量而不是引用——更安全的替代方案
    field_start: Option<usize>,
    field_end: Option<usize>,
    _pin: PhantomPinned,  // ← !Unpin 标记（2.11 Pin 上）
}

impl PinnedParser {
    /// 构造并立即 Pin 住
    pub fn new(json: String) -> Pin<Box<Self>> {
        let mut parser = Box::pin(PinnedParser {
            buffer: json,
            field_start: None,
            field_end: None,
            _pin: PhantomPinned,
        });

        // 在 Pin 住之后，指针可以安全地指向内部数据
        // （使用偏移量代替裸指针——更安全）
        let field_start = parser.buffer.find("\"name\":").map(|i| i + 8);
        let field_end = parser.buffer[field_start.unwrap_or(0)..]
            .find('"')
            .map(|i| field_start.unwrap() + i);

        // SAFETY: Pin 保证 parser 不会被移动
        unsafe {
            let mut_ref = Pin::as_mut(&mut parser).get_unchecked_mut();
            // Pin 投影：安全地访问被 Pin 保护的字段（2.12 Pin 下）
            mut_ref.field_start = field_start;
            mut_ref.field_end = field_end;
        }

        parser
    }

    /// 返回解析出的字段切片——精确指向 buffer 内部
    pub fn get_field<'a>(self: Pin<&'a Self>) -> Option<&'a str> {
        let this = self.get_ref();  // Pin 投影（2.12）
        match (this.field_start, this.field_end) {
            (Some(start), Some(end)) => Some(&this.buffer[start..end]),
            _ => None,
        }
    }
}

// ━━━ 使用 ━━━
fn demo_pinned_parser() {
    let mut parser = PinnedParser::new(r#"{"name":"Alice","age":30}"#.into());
    // parser 现在是 Pin<Box<PinnedParser>> —— 在堆上，不可移动

    // 安全地获取指向内部数据的引用
    let field = parser.as_ref().get_field();
    assert_eq!(field, Some("Alice"));

    // ✗ 不能移动：
    // let moved = *parser;  // 编译错误！Pin<Box<T>> 不实现 DerefMut
}
```

### 4.4 简易 Future —— 用 Union + ManuallyDrop 构建自引用状态机

```rust
/// ━━━ 手写 Future trait（简化版）━━━
///
/// 标准库的 Future 使用 Pin<&mut Self> 作为 poll 的参数
/// 原因：Future 通常是自引用状态机（由 async 块编译而来）
trait MyFuture {
    type Output;
    fn poll(self: Pin<&mut Self>, waker: &Waker) -> Poll<Self::Output>;
}

#[derive(Clone)]
struct Waker;  // 简化：标准库 Waker 更复杂

enum Poll<T> {
    Ready(T),
    Pending,
}

/// ━━━ 一个需要 Pin 的 Future：自引用状态机 ━━━
///
/// 这个 Future 内部持有的 buffer 和指向它的引用形成自引用
enum ReadFileState {
    Reading { buffer: Vec<u8>, bytes_read: usize },
    // ... 其他状态
}

struct ReadFileFuture {
    // ━━━ Union 的使用（2.8 Union）━━━
    // 多个状态共享同一块内存，节省空间
    // 标准库内部用类似机制实现 async 状态机
    state: ReadFileState,
}

impl MyFuture for ReadFileFuture {
    type Output = Vec<u8>;

    fn poll(mut self: Pin<&mut Self>, _waker: &Waker) -> Poll<Self::Output> {
        // Pin 保证 self 不会被移动，所以内部引用保持有效
        // 实际上 async 块编译出来的代码就是这样——Pin 引用的 self
        Poll::Pending  // 简化
    }
}

/// ━━━ Mini Executor ━━━
///
/// 职责：维护待 poll 的 future 列表，循环 poll 直到完成
struct MiniExecutor {
    // 标准库用 intrusive linked list + Pin<Box<dyn Future>>
    // 这里简化
    waker: Waker,
}

impl MiniExecutor {
    fn block_on<F: MyFuture>(&self, mut future: Pin<Box<F>>) -> F::Output {
        loop {
            match future.as_mut().poll(&self.waker) {
                Poll::Ready(output) => return output,
                Poll::Pending => {
                    // 真实世界会 yield 到事件循环
                    // 这里简化成自旋
                    std::thread::yield_now();
                }
            }
        }
    }
}

// ━━━ ManuallyDrop 在 Future 中的应用（2.4 ManuallyDrop）━━━
//
// Future 在 Pending 时不会被 drop——executor 持有它
// 当 executor 取消 future 时，才需要手动 drop
// 标准库用 ManuallyDrop 或类似机制来管理这个生命周期

use std::mem::ManuallyDrop;

struct CancelableFuture<F: MyFuture> {
    inner: ManuallyDrop<F>,  // 阻止自动 Drop
    cancelled: bool,
}

impl<F: MyFuture> Drop for CancelableFuture<F> {
    fn drop(&mut self) {
        if !self.cancelled {
            // SAFETY: 我们确保只在 drop 时手动调用
            unsafe { ManuallyDrop::drop(&mut self.inner); }
        }
        // 如果 cancelled，inner 不会被 drop——调用者负责清理
    }
}
```

## 5. Demo 4: Arena/Bump 分配器 —— 批量分配，批量释放

### 5.1 设计哲学

> "不是每个分配都需要 `malloc`。在编译器和游戏引擎中，生命周期相同的对象被批量分配、批量释放——一个简单的 bump 指针比通用分配器快一个数量级。理解 Arena 就是理解'为什么 Rust 允许你绕过全局分配器'。"

### 5.2 BumpAllocator —— 最简单的 Arena

```rust
use std::alloc::{self, Layout};
use std::cell::Cell;
use std::ptr::NonNull;

/// ━━━ Bump Allocator: 偏移指针 + 大块内存 ━━━
///
/// 分配策略：维护一块大内存和一个偏移量。
/// 每次分配：返回 `base + offset`，offset 增加。
/// 释放：所有分配同时失效（批量释放）。
struct BumpAllocator {
    memory: NonNull<u8>,
    capacity: usize,
    offset: Cell<usize>,  // 当前偏移（Cell 允许通过 &self 修改）
}

impl BumpAllocator {
    /// 创建指定容量的 bump allocator
    pub fn new(capacity: usize) -> Self {
        // ━━━ Layout 计算对齐（2.5 Layout/Alignment）━━━
        let layout = Layout::from_size_align(capacity, std::mem::align_of::<usize>())
            .expect("invalid layout");

        // SAFETY: layout 的 size > 0（capacity > 0）
        let memory = unsafe {
            let ptr = alloc::alloc(layout);
            NonNull::new_unchecked(ptr)
        };

        BumpAllocator {
            memory,
            capacity,
            offset: Cell::new(0),
        }
    }

    /// 分配指定 Layout 的内存
    ///
    /// 返回指向分配空间的指针，或 None（空间不足）
    pub fn allocate(&self, layout: Layout) -> Option<NonNull<u8>> {
        let current_offset = self.offset.get();

        // ━━━ 对齐计算（2.5 Alignment）━━━
        // 确保返回的指针对齐到 layout.align()
        let align_offset = (current_offset + layout.align() - 1) & !(layout.align() - 1);

        let new_offset = align_offset + layout.size();

        if new_offset <= self.capacity {
            self.offset.set(new_offset);
            // SAFETY: memory 有效，align_offset 在 capacity 内
            unsafe { Some(NonNull::new_unchecked(self.memory.as_ptr().add(align_offset))) }
        } else {
            None
        }
    }

    /// 重置分配器——所有之前的分配都失效
    pub fn reset(&self) {
        self.offset.set(0);
    }

    /// 当前已分配字节数
    pub fn allocated(&self) -> usize {
        self.offset.get()
    }
}

impl Drop for BumpAllocator {
    fn drop(&mut self) {
        // ━━━ 注意：Bump 分配器不调用每个分配的 Drop ━━━
        // 这意味着分配在 arena 中的类型必须：
        // 1. 不实现 Drop，或
        // 2. 调用者手动管理 drop
        let layout = Layout::from_size_align(self.capacity, std::mem::align_of::<usize>())
            .unwrap();
        unsafe { alloc::dealloc(self.memory.as_ptr(), layout) };
    }
}
```

### 5.3 `TypedArena<T>` —— 类型安全的 Arena

```rust
use std::marker::PhantomData;

/// ━━━ TypedArena: 只分配一种类型的 Arena ━━━
///
/// 相比 BumpAllocator 的优势：
/// - 类型安全：只存储 T 类型
/// - 自动计算 Layout
/// - PhantomData 标注所有权（2.6 PhantomData）
pub struct TypedArena<T> {
    allocator: BumpAllocator,
    // PhantomData 告诉编译器：TypedArena 拥有它所分配的 T
    _marker: PhantomData<Vec<T>>,
}

impl<T> TypedArena<T> {
    pub fn with_capacity(capacity_in_bytes: usize) -> Self {
        TypedArena {
            allocator: BumpAllocator::new(capacity_in_bytes),
            _marker: PhantomData,
        }
    }

    /// 在 Arena 中分配一个 T，返回可变引用
    pub fn alloc(&self, value: T) -> Option<&mut T> {
        let layout = Layout::new::<T>();

        // ━━━ ZST 特殊处理（2.9 ZST）━━━
        // ZST 不需要实际分配空间——返回一个"假"引用
        if layout.size() == 0 {
            // ZST: 返回 dangling 指针的引用（因为读/写 0 字节总是安全的）
            return Some(unsafe { &mut *NonNull::<T>::dangling().as_ptr() });
        }

        let ptr = self.allocator.allocate(layout)?;
        // SAFETY: ptr 正确对齐，空间足够（layout.size() 字节）
        unsafe { ptr.cast::<T>().write(value) };
        Some(unsafe { &mut *ptr.cast::<T>().as_ptr() })
    }

    /// 批量分配——从迭代器收集到 Arena
    pub fn alloc_batch<I: IntoIterator<Item = T>>(&self, iter: I) -> Option<&mut [T]> {
        let items: Vec<T> = iter.into_iter().collect();
        let count = items.len();
        let layout = Layout::array::<T>(count).ok()?;

        if layout.size() == 0 {
            // ZST batch——同样不需要空间
            return Some(&mut []);
        }

        let ptr = self.allocator.allocate(layout)?;
        for (i, item) in items.into_iter().enumerate() {
            unsafe { ptr.cast::<T>().as_ptr().add(i).write(item) };
        }
        Some(unsafe { std::slice::from_raw_parts_mut(ptr.cast::<T>().as_ptr(), count) })
    }

    pub fn reset(&self) {
        // ━━━ 注意：reset 不调用 Drop ━━━
        // Arena 中的值没有被正确释放——适用于不实现 Drop 的类型
        self.allocator.reset();
    }
}

// ━━━ 使用示例 ━━━
fn demo_arena() {
    let arena = TypedArena::<String>::with_capacity(1024);

    let s1 = arena.alloc(String::from("hello")).unwrap();
    let s2 = arena.alloc(String::from("world")).unwrap();

    assert_eq!(*s1, "hello");
    *s2 = String::from("rust");
    assert_eq!(*s2, "rust");

    // 批量重置——s1 和 s2 失效
    // （注意：String 的 Drop 没有被调用——内存泄漏！
    //  这展示了 arena 的局限性）
    // arena.reset();
}
```

### 5.4 Arena + DropArena —— 支持 Drop 的 Arena

```rust
/// ━━━ DropArena: 记录所有分配的指针，重置时逐个 Drop ━━━
///
/// 权衡：比纯 BumpAllocator 慢（需要遍历），但安全（不会泄漏 Drop 类型）
struct DropArena {
    allocator: BumpAllocator,
    // 记录需要 Drop 的指针和对应的 Layout
    drop_list: std::cell::RefCell<Vec<(NonNull<u8>, Layout)>>,
}

impl DropArena {
    pub fn allocate_with_drop<T>(&self, value: T) -> Option<&mut T> {
        let layout = Layout::new::<T>();
        let ptr = self.allocator.allocate(layout)?;

        unsafe { ptr.cast::<T>().write(value) };

        // 记录以便后续 drop
        self.drop_list.borrow_mut().push((ptr, layout));

        Some(unsafe { &mut *ptr.cast::<T>().as_ptr() })
    }

    /// 重置：按 LIFO 顺序 drop 所有分配，然后重置 bump
    pub fn reset(&self) {
        // ━━━ Drop 顺序：后分配的先释放（2.4 Drop 顺序）━━━
        let mut drop_list = self.drop_list.borrow_mut();
        while let Some((ptr, layout)) = drop_list.pop() {
            unsafe { std::ptr::drop_in_place(ptr.as_ptr()) };
            // 注意：这里不释放内存——内存由 bump allocator 统一管理
        }
        self.allocator.reset();
    }
}
```

### 6.4 参考答案

```rust
use std::marker::{PhantomData, PhantomPinned};
use std::mem::MaybeUninit;
use std::pin::Pin;
use std::ptr::NonNull;
use std::cell::Cell;
use std::ops::{Deref, DerefMut};

pub struct ObjectPool<T, const N: usize> {
    // 槽位存储——MaybeUninit 数组（2.10 MaybeUninit）
    slots: [MaybeUninit<T>; N],
    // 位图：true = 已占用，false = 空闲
    occupied: [Cell<bool>; N],
}

pub struct PoolRef<'pool, T> {
    // 指向池中某个槽位
    ptr: NonNull<T>,           // NonNull 提供 Niche 优化（2.13）
    // 回指池——用于释放时标记槽位为空闲
    occupied_slot: &'pool Cell<bool>,
    // 生命周期标注（2.6 PhantomData）
    _marker: PhantomData<&'pool mut T>,
    // 支持 Pin（2.11）
    _pin: PhantomPinned,
}

impl<T, const N: usize> ObjectPool<T, N> {
    pub fn new() -> Self {
        // SAFETY: MaybeUninit 数组可以未初始化
        ObjectPool {
            slots: unsafe { MaybeUninit::uninit().assume_init() },
            occupied: [const { Cell::new(false) }; N],  // 需要 const 表达式支持
        }
    }

    pub fn alloc(&self, value: T) -> Option<PoolRef<'_, T>> {
        // ━━━ ZST 处理（2.9）━━━
        if std::mem::size_of::<T>() == 0 {
            // ZST: 不需要实际槽位——返回 dangling 指针
            // 简化处理：永远返回"成功"
            // （真实实现中会用单独的计数追踪 ZST）
            todo!("ZST handling in object pool")
        }

        // 找到第一个空闲槽位
        for i in 0..N {
            if !self.occupied[i].get() {
                // 占用槽位
                self.occupied[i].set(true);

                // 写入值到 MaybeUninit 槽位
                let slot = &self.slots[i] as *const MaybeUninit<T> as *mut T;
                unsafe { slot.write(value) };

                return Some(PoolRef {
                    ptr: unsafe { NonNull::new_unchecked(slot) },
                    occupied_slot: &self.occupied[i],
                    _marker: PhantomData,
                    _pin: PhantomPinned,
                });
            }
        }

        None  // 池已满
    }
}

impl<T> Deref for PoolRef<'_, T> {
    type Target = T;
    fn deref(&self) -> &T {
        unsafe { self.ptr.as_ref() }
    }
}

impl<T> DerefMut for PoolRef<'_, T> {
    fn deref_mut(&mut self) -> &mut T {
        unsafe { self.ptr.as_mut() }
    }
}

impl<T> Drop for PoolRef<'_, T> {
    fn drop(&mut self) {
        // 归还槽位到池中
        unsafe { std::ptr::drop_in_place(self.ptr.as_ptr()) };
        self.occupied_slot.set(false);
    }
}

// ━━━ 型变（2.6）━━━
// ObjectPool<T, N> 对 T 是协变的——因为 PhantomData<T> 默认协变
// PoolRef<'pool, T> 对 'pool 和 T 是协变的——编译器自动推导

// ━━━ 不支持 Copy/Clone —— ObjectPool 是独占资源 ━━━
// 如果需要，可以实现 Clone for PoolRef（允许多个引用指向同一槽位）
// 但这需要引用计数机制（类似 Rc）
```
